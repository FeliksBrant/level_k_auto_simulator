255  float maxDiff = 0.3;
256 
257  for (i = 0; i < _JS_MAX_AXES * NUM_JOY; i++) {

267 static void Idle(void)
268 {
269  int mask;
270  int b, i;
271  int index;
272  const char *str;
273  int axis;
274 
275  GfctrlMouseGetCurrent(&mouseInfo);
276 
277  /* Check for a mouse button pressed */
278  for (i = 0; i < 3; i++) {
279  if (mouseInfo.edgedn[i]) {
280  glutIdleFunc(GfuiIdle);
281  InputWaited = 0;
282  str = GfctrlGetNameByRef(GFCTRL_TYPE_MOUSE_BUT, i);
283  CurrentCmd->ref.index = i;
284  CurrentCmd->ref.type = GFCTRL_TYPE_MOUSE_BUT;
285  GfuiButtonSetText (scrHandle, CurrentCmd->Id, str);
286  glutPostRedisplay();
287  return;
288  }
289  }
290 
291  /* Check for a mouse axis moved */
292  for (i = 0; i < 4; i++) {
293  if (mouseInfo.ax[i] > 20.0) {
294  glutIdleFunc(GfuiIdle);
295  InputWaited = 0;
296  str = GfctrlGetNameByRef(GFCTRL_TYPE_MOUSE_AXIS, i);
297  CurrentCmd->ref.index = i;
298  CurrentCmd->ref.type = GFCTRL_TYPE_MOUSE_AXIS;
299  GfuiButtonSetText (scrHandle, CurrentCmd->Id, str);
300  glutPostRedisplay();
301  return;
302  }
303  }
304 
305  /* Check for a Joystick button pressed */
306  for (index = 0; index < NUM_JOY; index++) {
307  if (js[index]) {
308  js[index]->read(&b, &ax[index * _JS_MAX_AXES]);
309 
310  /* Joystick buttons */
311  for (i = 0, mask = 1; i < 32; i++, mask *= 2) {
312  if (((b & mask) != 0) && ((rawb[index] & mask) == 0)) {
313  /* Button i fired */
314  str = GfctrlGetNameByRef(GFCTRL_TYPE_JOY_BUT, i + 32 * index);
315  if (!GfctrlIsEventBlacklisted(PrefHdle, CurrentSection, str)) {
316  glutIdleFunc(GfuiIdle);
317  InputWaited = 0;
318  CurrentCmd->ref.index = i + 32 * index;
319  CurrentCmd->ref.type = GFCTRL_TYPE_JOY_BUT;
320  GfuiButtonSetText (scrHandle, CurrentCmd->Id, str);
321  glutPostRedisplay();
322  rawb[index] = b;
323  return;
324  }
325  }
326  }
327  rawb[index] = b;
328  }
329  }
330 
331  /* detect joystick movement */
332  axis = getMovedAxis();
333  if (axis != -1) {
334  glutIdleFunc(GfuiIdle);
335  InputWaited = 0;
336  CurrentCmd->ref.type = GFCTRL_TYPE_JOY_AXIS;
337  CurrentCmd->ref.index = axis;
338  str = GfctrlGetNameByRef(GFCTRL_TYPE_JOY_AXIS, axis);
339  GfuiButtonSetText (scrHandle, CurrentCmd->Id, str);
340  glutPostRedisplay();
341  return;
342  }
343 }
