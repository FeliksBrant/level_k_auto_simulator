65 void Complex::finish(int n, const Polytope **p) {
66  proceed();
67  leaves = new BBoxLeaf[n];
68  count = n;
69  for (int i = 0; i < n; ++i) new(&leaves[i]) BBoxLeaf(p[i]);
70  if (n >= 2) {
71  free_node = new BBoxInternal[n-1];
72  root = free_node;
73  new(free_node++) BBoxInternal(n, leaves);
74  }
