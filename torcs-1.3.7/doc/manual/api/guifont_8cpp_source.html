49 void swap32(unsigned int *p, unsigned int size)
50 {
51  unsigned int i, t;
52  for (i = 0; i < size; i += 4) {
53  t = (unsigned int) *p;
54  *p = (t & 0xff000000U) >> 24;
55  *p |= (t & 0x00ff0000U) >> 8;
56  *p |= (t & 0x0000ff00U) << 8;
57  *p |= (t & 0x000000ffU) << 24;
58  p++;
59  }
60 }

104 GfuiFontClass::GfuiFontClass(char *FileName)
105 {
106  FILE *Input;
107  char *TexBytes;
108  int Num;
109  uint Tex;
110 
111  font = NULL;
112  size = 8.0;
113 
114  //Open font file
115  if ((Input = fopen(FileName, "rb")) == NULL) {
116  perror(FileName);
117  return;
118  }
119 
120  if ((font = (GLFONT *)malloc(sizeof(GLFONT))) == NULL) {
121  return;
122  }
123 
124  //Read glFont structure
125  //fread(font, sizeof(GLFONT), 1, Input);
126  fread(font, 24, 1, Input); // for IA64...
127 
128 #ifndef WIN32
129 #if BYTE_ORDER == BIG_ENDIAN
130  swap32((unsigned int *) font, 24);
131 #endif
132 #endif
133 
134  //Get number of characters
135  Num = font->IntEnd - font->IntStart + 1;
136 
137  //Allocate memory for characters
138  if ((font->Char = (GLFONTCHAR *)malloc(sizeof(GLFONTCHAR) * Num)) == NULL) {
139  free(font);
140  font = NULL;
141  fclose(Input);
142  return;
143  }
144 
145  //Read glFont characters
146  fread(font->Char, sizeof(GLFONTCHAR), Num, Input);
147 
148 #ifndef WIN32
149 #if BYTE_ORDER == BIG_ENDIAN
150  swap32((unsigned int *) font->Char, sizeof(GLFONTCHAR) * Num);
151 #endif
152 #endif
153 
154  //Get texture size
155  Num = font->TexWidth * font->TexHeight * 2;
156 
157  //Allocate memory for texture data
158  if ((TexBytes = (char *)malloc(Num)) == NULL) {
159  fclose(Input);
160  return;
161  }
162 
163  //Read texture data
164  fread(TexBytes, sizeof(char), Num, Input);
165 
166  fclose(Input);
167 
168  //Save texture number
169  glGenTextures(1, &Tex);
170  font->Tex = Tex;
171  //Set texture attributes
172  glBindTexture(GL_TEXTURE_2D, Tex);
173  glTexParameterf(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP);
174  glTexParameterf(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP);
175  glTexParameterf(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
176  glTexParameterf(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
177  glTexEnvf(GL_TEXTURE_ENV, GL_TEXTURE_ENV_MODE, GL_MODULATE);
178 
179  //Create texture
180  glTexImage2D(GL_TEXTURE_2D, 0, 2, font->TexWidth,
181  font->TexHeight, 0, GL_LUMINANCE_ALPHA,
182  GL_UNSIGNED_BYTE, (void *)TexBytes);
183 
184  //Clean up
185  free(TexBytes);
186 
187  //Return pointer to new font
188  return;
189 }

249 void GfuiFontClass::output(int X, int Y, const char* text)
250 {
251  int Length, i;
252  GLFONTCHAR *Char;
253  float x = (float)X;
254  float y = (float)Y;
255 
256  //Return if we don't have a valid glFont
257  if (font == NULL) return;
258 
259  //Get length of string
260  Length = strlen(text);
261 
262  //Begin rendering quads
263  glBindTexture(GL_TEXTURE_2D, font->Tex);
264  glBegin(GL_QUADS);
265 
266  //Loop through characters
267  for (i = 0; i < Length; i++)
268  {
269  //Get pointer to glFont character
270  Char = &font->Char[(int)text[i] - font->IntStart];
271 
272  //Specify vertices and texture coordinates
273  glTexCoord2f(Char->tx1, Char->ty1);
274  glVertex2f(x, y + Char->dy * size);
275  glTexCoord2f(Char->tx1, Char->ty2);
276  glVertex2f(x, y);
277  glTexCoord2f(Char->tx2, Char->ty2);
278  glVertex2f(x + Char->dx * size, y);
279  glTexCoord2f(Char->tx2, Char->ty1);
280  glVertex2f(x + Char->dx * size, y + Char->dy * size);
281 
282  //Move to next character
283  x += Char->dx*size;
284  }
285 
286  //Stop rendering quads
287  glEnd();
288 }
