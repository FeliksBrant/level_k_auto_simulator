117 void CarSoundData::calculateAttenuation (tCarElt* car)
118 {
119  if (car->_state & RM_CAR_STATE_NO_SIMU) {
120  attenuation = 0.0f;
121  return;
122  }
123  float d = 0.0;
124  for (int i=0; i<3; i++) {

133 void CarSoundData::calculateEngineSound (tCarElt* car)
134 {
135  float mpitch = base_frequency * (float)(car->_enginerpm) / 600.0;
136  engine.f = mpitch;
137  engine.a = 1.0f;
138  if (car->_state & RM_CAR_STATE_NO_SIMU) {
139  engine.a = 0.0f;
140  engine.lp = 1.0;
141  turbo.a = 0.0f;
142  turbo.f = 1.0f;
143  return;
144  }
145  //assert(car->index == eng_pri.id);
146 
147  float gear_ratio = car->_gearRatio[car->_gear + car->_gearOffset];
148  axle.a = 0.2f*(tanh(100.0f*(fabs(pre_axle - mpitch))));
149  axle.f = (pre_axle + mpitch)*0.05f*fabs(gear_ratio);
150  pre_axle = (pre_axle + mpitch)*.5;
151 
152  if (turbo_on) {
153  float turbo_target = 0.1f;
154  float turbo_target_vol = 0.0f;
155  if (car->_enginerpm > turbo_rpm) {
156  turbo_target = 0.1f + 0.9f * smooth_accel;
157  turbo_target_vol = 0.1f * smooth_accel;
158  }
159 
160  turbo.a += 0.1f * (turbo_target_vol - turbo.a) * (0.1f + smooth_accel);
161  float turbo_target_pitch = turbo_target * car->_enginerpm / 600.0f;
162  turbo.f += turbo_ilag * (turbo_target_pitch - turbo.f) * (smooth_accel);
163  turbo.f -= turbo.f * 0.01f * (1.0-smooth_accel);//.99f;
164  } else {
165  turbo.a = 0.0;
166  }
167  smooth_accel = smooth_accel*0.5f + 0.5*(car->ctrl.accelCmd*.99f+0.01f);
168 
169  // engine filter proportional to revcor2.
170  // when accel = 0, lp \in [0.0, 0.25]
171  // when accel = 1, lp \in [0.25, 1.0]
172  // interpolate linearly for values in between.
173  float rev_cor = car->_enginerpm/car->_enginerpmRedLine;
174  float rev_cor2 = rev_cor * rev_cor;
175  engine.lp = (0.75f*rev_cor2 + 0.25f)*smooth_accel
176  + (1.0f-smooth_accel)*0.25f*rev_cor2;
177 
178  // TODO: filter for exhaust and car body resonance?
179 
180 }

184 void CarSoundData::calculateBackfireSound (tCarElt* car)
185 {
186  if (car->_state & RM_CAR_STATE_NO_SIMU) {

198 void CarSoundData::calculateTyreSound(tCarElt* car)
199 {
200  grass_skid.a = 0.0;
201  grass.a = 0.0;
202  grass.f = 1.0f;
203  road.a = 0.0;
204  road.f = 0.0f;
205  bool flag = false;
206  int i;
207  for (i = 0; i<4; i++) {
208  wheel[i].skid.a = 0.0f;
209  wheel[i].skid.f = 1.0f;
210  }
211  if (car->_state & RM_CAR_STATE_NO_SIMU) {
212  return;
213  }
214 
215  for (i = 0; i<4; i++) {
216  if (car->_wheelSpinVel(i) > 0.1f) {
217  flag = true;
218  break;
219  }
220  }
221 
222  if (
223  (car->_state & RM_CAR_STATE_NO_SIMU) ||
224  ((car->pub.speed < 0.3f) &&
225  (flag == false))
226  ) {
227  return;
228  }
229 
230  for (i = 0; i<4; i++) {
231  const char* s = NULL;
232  tdble roughness = 0.0f;
233  tdble roughnessFreq = 1.0f;
234  float ride = 0.0001f;
235  float tmpvol = car->pub.speed*0.01f;
236  if (car==NULL) {
237  fprintf (stderr, "Error: (grsound.c) no car\n");
238  continue;
239  } else if (car->priv.wheel==NULL) {
240  fprintf (stderr, "Error: (grsound.c) no wheels\n");
241  continue;
242  } else if (car->priv.wheel[i].seg==NULL) {
243  fprintf (stderr, "Error: (grsound.c) no seg\n");
244  continue;
245  } else if (car->priv.wheel[i].seg->surface==NULL) {
246  fprintf (stderr, "Error: (grsound.c) no surface\n");
247  continue;
248  } else if (car->priv.wheel[i].seg->surface->material==NULL) {
249  fprintf (stderr, "Error: (grsound.c) no material\n");
250  continue;
251  } else {
252  s = car->priv.wheel[i].seg->surface->material;
253  roughness = car->priv.wheel[i].seg->surface->kRoughness;
254  roughnessFreq = 2.0f*EX_PI * car->priv.wheel[i].seg->surface->kRoughWaveLen;
255  if (roughnessFreq>2.0f) {
256  roughnessFreq = 2.0f + tanh(roughnessFreq-2.0f);
257  }
258  //ride = car->priv.wheel[i].rollRes;
259  ride = 0.001f * car->_reaction[i];
260  }
261 
262  int out_of_road = false;
263 
264  if ((s)
265  &&((strcmp(s, TRK_VAL_GRASS)==0)
266  ||(strcmp(s, TRK_VAL_SAND)==0)
267  ||(strcmp(s, TRK_VAL_DIRT)==0)
268  ||(strstr(s, "sand"))
269  ||(strstr(s, "dirt"))
270  ||(strstr(s, "grass"))
271  ||(strstr(s, "gravel"))
272  ||(strstr(s, "mud"))
273  )) {
274  out_of_road = true;
275  }
276 
277  wheel[i].skid.a = 0.0f;
278  wheel[i].skid.f = 1.0f;
279 
280  if (out_of_road==false) {
281  float tmppitch = tmpvol*(0.75f+0.25f*roughnessFreq);
282  float wind_noise = 1.0f;
283  float road_noise = 0.25f;
284  tmpvol = tmpvol*(wind_noise + ride*road_noise);
285  if (road.a < tmpvol) {
286  road.a = tmpvol;
287  road.f = tmppitch;
288  }
289 
290  if (car->_skid[i] > 0.05f) {
291  //skvol[i] = (float)car->_skid[i];
292  //skpitch[i] = 0.7+0.3*roughnessFreq;
293  wheel[i].skid.a = (float)car->_skid[i]-0.05f;
294  float wsa = tanh((car->_wheelSlipAccel(i)+10.0f)*0.01f);
295  wheel[i].skid.f = (0.3f - 0.3f*wsa + 0.3f*roughnessFreq)/(1.0f+0.5f*tanh(car->_reaction[i]*0.0001f));
296  } else {
297  wheel[i].skid.a = 0.0f;
298  wheel[i].skid.f = 1.0f;
299  }
300  //printf ("%d %f %f\n", i, wheel[i].skid.a, wheel[i].skid.f);
301  } else {
302  float tmppitch = tmpvol*(0.5f + 0.5f*roughnessFreq);
303 
304  tmpvol = (0.5f+0.2f*tanh(0.5f*roughness))*tmpvol * ride;
305 
306  if (grass.a < tmpvol) {
307  grass.a = tmpvol;
308  grass.f = tmppitch;
309  }
310  if (grass_skid.a < car->_skid[i]) {
311  grass_skid.a = (float) car->_skid[i];
312  grass_skid.f = 1.0f;
313  }
314  }
315 
316  }
317 
318  for (i = 0; i<4; i++) {
319  tdble az = car->_yaw;
320  tdble Sinz = sin(az);
321  tdble Cosz = cos(az);
322 
323  tdble x = car->priv.wheel[i].relPos.x;
324  tdble y = car->priv.wheel[i].relPos.y;
325 
326  tdble dx = x * Cosz - y * Sinz;
327  tdble dy = x * Sinz + y * Cosz;
328 
329  tdble dux = -car->_yaw_rate * y;
330  tdble duy = car->_yaw_rate * x;
331 
332  dux = dux * Cosz - duy * Sinz;
333  duy = dux * Sinz + duy * Cosz;
334 
335  wheel[i].u[0] = car->pub.DynGCg.vel.x + dux;
336  wheel[i].u[1] = car->pub.DynGCg.vel.y + duy;
337  wheel[i].u[2] = car->pub.DynGCg.vel.z;
338  wheel[i].p[0] = car->pub.DynGCg.pos.x + dx;
339  wheel[i].p[1] = car->pub.DynGCg.pos.y + dy;
340  wheel[i].p[2] = car->pub.DynGCg.pos.z;
341  }
342 }

363  return;
364  }
365 
366 
367 
368  int collision = car->priv.collision;
369  if (collision) {
370  if (collision & 1) {
371  skid_metal.a = car->pub.speed*0.01;
372  skid_metal.f = .5+0.5*skid_metal.a;
373  drag_collision.f = skid_metal.f;
374  } else {
375  skid_metal.a = 0;
376  }
377 
378  if ((collision & 16 )) {
379  bottom_crash = true;
380  }
381 
382  if ((collision & 8 )) {
383  bang = true;
384  }
385 
386  if (((collision & 1) ==0) ||
387  ((collision & 2)
388  &&(skid_metal.a >drag_collision.a))) {
389  crash = true;
390  }
391  car->priv.collision = 0;
392  }
393 
394  drag_collision.a = 0.9f*drag_collision.a + skid_metal.a;
395  if (drag_collision.a>1.0f) {
396  drag_collision.a = 1.0f;
397  }
398  skid_metal.a = drag_collision.a;
399  skid_metal.f = drag_collision.f;
