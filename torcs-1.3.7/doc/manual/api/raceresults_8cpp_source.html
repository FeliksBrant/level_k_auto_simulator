52 /*
53  Applies pending penalties after the race:
54  - Time penalties
55  - Pending stop & go (can happen if penalty is given within the last 5 laps)
56  - Pending drive through (dito)
57 
58  I assume that the average track speed is 300km/h (v0), and that the pit entry
59  and exit costs 10s. We call the pit speed limit v1, and the length of the pit
60  lane s. So the time lost with driving through the pit lane is
61  dt = s*(v0-v1)/(v0*v1). We assume that stopping costs additional 6 seconds.
62 
63  Drive through: 10 + dt [s]
64  Stop & Go: 10 + 6 + dt [s]
65 
66  Applying time penalties with cars in different laps is problematic, picture
67  this situation: We have 3 cars running, pos 1 car is right behind the pos 3
68  car (so it will overlap in a moment), pos 3 car is right behind pos 2 car,
69  and pos 2 car has a 30s penalty pending. Now right before the finish line the
70  pos 1 car passes (overlaps) the pos 3 car, so the race is done for pos 1 car,
71  and a moment later for the pos 3 car. But the pos 2 car can finish its last
72  lap, although it has a 30s penalty pending, which would have thrown it behind
73  the pos 3 car, if the overlapping not had happened.
74 
75  So just adding times is not enough (btw. the FIA does just add times... weird).
76 
77  All time penalties are added up and then applied to the ranking like this:
78  We "pump up" the result of the opponents with less laps up to our laps
79  with a simple linear model: opponent_race_time/opponent_laps*our_race_laps, so
80  we are worse when:
81  our_race_time + our_penalty_time >
82  opponent_race_time/opponent_laps*our_race_laps + opponent_penalty_time
83 
84  This works as well for opponents in the same lap, opponent_laps*our_race_laps is
85  then 1. The calculation requires that at least one full lap has been driven by
86  the drivers which are compared.
87 
88  Wrecked cars are considered worse than a penalty.
89 */
90 static void ReApplyRaceTimePenalties(void)
91 {
92  // First update all penalty times, apply pending drive through/stop and go
93  int i;
94  tSituation *s = ReInfo->s;
95  tCarPenalty *penalty;
96  tCarElt* car;
97 
98  if (ReInfo->track->pits.type == TR_PIT_ON_TRACK_SIDE) {
99  const tdble drivethrough = 10.0f;
100  const tdble stopandgo = 6.0f + drivethrough;
101 
102  const tdble v0 = 84.0f;
103  tdble v1 = ReInfo->track->pits.speedLimit;
104  tdble dv = v0 - v1;
105  tdble dt = 0.0;
106 
107  if (dv > 1.0f && v1 > 1.0f) {
108  dt = (ReInfo->track->pits.nMaxPits*ReInfo->track->pits.len)*dv/(v0*v1);
109  }
110 
111  for (i = 0; i < s->_ncars; i++) {
112  car = s->cars[i];
113  penalty = GF_TAILQ_FIRST(&(car->_penaltyList));
114  while (penalty) {
115  if (penalty->penalty == RM_PENALTY_DRIVETHROUGH) {
116  car->_penaltyTime += dt + drivethrough;
117  } else if (penalty->penalty == RM_PENALTY_STOPANDGO) {
118  car->_penaltyTime += dt + stopandgo;
119  } else {
120  GfError("Unknown penalty.");
121  }
122  penalty = GF_TAILQ_NEXT(penalty, link);
123  }
124  }
125  }
126 
127  // Now sort the cars taking into account the penalties
128  int j;
129  for (i = 1; i < s->_ncars; i++) {
130  j = i;
131  while (j > 0) {
132  // Order without penalties is already ok, so if there is no penalty we do not move down
133  if (s->cars[j-1]->_penaltyTime > 0.0f) {
134  int l1 = MIN(s->cars[j-1]->_laps, s->_totLaps + 1) - 1;
135  int l2 = MIN(s->cars[j]->_laps, s->_totLaps + 1) - 1;
136  // If the drivers did not at least complete one lap we cannot apply the rule, check.
137  // If the cars are wrecked we do not care about penalties.
138  if (
139  l1 < 1 ||
140  l2 < 1 ||
141  (s->raceInfo.maxDammage < s->cars[j-1]->_dammage) ||
142  (s->raceInfo.maxDammage < s->cars[j]->_dammage))
143  {
144  // Because the cars came already presorted, all following cars must be even worse,
145  // so we can break the iteration here.
146  i = s->_ncars; // Break outer loop
147  break; // Break inner loop
148  }
149 
150  tdble t1 = s->cars[j-1]->_curTime + s->cars[j-1]->_penaltyTime;
151  tdble t2 = s->cars[j]->_curTime*tdble(l1)/tdble(l2) + s->cars[j]->_penaltyTime;
152 
153  if (t1 > t2) {
154  // Swap
155  car = s->cars[j];
156  s->cars[j] = s->cars[j-1];
157  s->cars[j-1] = car;
158  s->cars[j]->_pos = j+1;
159  s->cars[j-1]->_pos = j;
160  j--;
161  continue;
162  }
163  }
164  j = 0;
165  }
166  }
167 }

219 void
220 ReUpdateStandings(void)
221 {
222  int maxDrv;
223  int curDrv;
224  int runDrv;
225  char *modName;
226  int drvIdx;
227  int points;
228  int i, j;
229  int found;
230  tReStandings *standings = 0;
231  void *results = ReInfo->results;
232  const int BUFSIZE = 1024;
233  char str1[BUFSIZE], str2[BUFSIZE], path[BUFSIZE], path2[BUFSIZE];
234 
235  snprintf(path, BUFSIZE, "%s/%s/%s/%s", ReInfo->track->name, RE_SECT_RESULTS, ReInfo->_reRaceName, RE_SECT_RANK);
236 
237  runDrv = GfParmGetEltNb(results, path);
238  curDrv = GfParmGetEltNb(results, RE_SECT_STANDINGS);
239  maxDrv = curDrv + runDrv;
240 
241  standings = (tReStandings *)calloc(maxDrv, sizeof(tReStandings));
242 
243  /* Read the current standings */
244  for (i = 0; i < curDrv; i++) {
245  snprintf(path2, BUFSIZE, "%s/%d", RE_SECT_STANDINGS, i + 1);
246  standings[i].carName = strdup(GfParmGetStr(results, path2, RE_ATTR_NAME, 0));
247  standings[i].modName = strdup(GfParmGetStr(results, path2, RE_ATTR_MODULE, 0));
248  standings[i].drvIdx = (int)GfParmGetNum(results, path2, RE_ATTR_IDX, NULL, 0);
249  standings[i].points = (int)GfParmGetNum(results, path2, RE_ATTR_POINTS, NULL, 0);
250  }
251 
252  GfParmListClean(results, RE_SECT_STANDINGS);
253 
254  for (i = 0; i < runDrv; i++) {
255  /* Search the driver in the standings */
256  found = 0;
257  snprintf(path, BUFSIZE, "%s/%s/%s/%s/%d", ReInfo->track->name, RE_SECT_RESULTS, ReInfo->_reRaceName, RE_SECT_RANK, i + 1);
258  const char* carName = GfParmGetStr(results, path, RE_ATTR_NAME, 0);
259  for (j = 0; j < curDrv; j++) {
260  if (!strcmp(carName, standings[j].carName)) {
261  found = 1;
262  break;
263  }
264  }
265 
266  if (!found) {
267  /* Add the new driver */
268  curDrv++;
269  standings[j].carName = strdup(carName);
270  standings[j].modName = strdup(GfParmGetStr(results, path, RE_ATTR_MODULE, 0));
271  standings[j].drvIdx = (int)GfParmGetNum(results, path, RE_ATTR_IDX, NULL, 0);
272  standings[j].points = (int)GfParmGetNum(results, path, RE_ATTR_POINTS, NULL, 0);
273  } else {
274  /* Add the new points */
275  standings[j].points += (int)GfParmGetNum(results, path, RE_ATTR_POINTS, NULL, 0);
276  }
277  /* bubble sort... */
278  while (j > 0) {
279  if (standings[j - 1].points >= standings[j].points) {
280  break;
281  }
282  /* Swap with preceeding */
283  char* tmpCarName;
284  tmpCarName = standings[j].carName;
285  modName = standings[j].modName;
286  drvIdx = standings[j].drvIdx;
287  points = standings[j].points;
288 
289  standings[j].carName = standings[j - 1].carName;
290  standings[j].modName = standings[j - 1].modName;
291  standings[j].drvIdx = standings[j - 1].drvIdx;
292  standings[j].points = standings[j - 1].points;
293 
294  standings[j - 1].carName = tmpCarName;
295  standings[j - 1].modName = modName;
296  standings[j - 1].drvIdx = drvIdx;
297  standings[j - 1].points = points;
298 
299  j--;
300  }
301  }
302 
303  /* Store the standing back */
304  for (i = 0; i < curDrv; i++) {
305  snprintf(path, BUFSIZE, "%s/%d", RE_SECT_STANDINGS, i + 1);
306  GfParmSetStr(results, path, RE_ATTR_NAME, standings[i].carName);
307  free(standings[i].carName);
308  GfParmSetStr(results, path, RE_ATTR_MODULE, standings[i].modName);
309  free(standings[i].modName);
310  GfParmSetNum(results, path, RE_ATTR_IDX, NULL, standings[i].drvIdx);
311  GfParmSetNum(results, path, RE_ATTR_POINTS, NULL, standings[i].points);
312  }
313  free(standings);
314 
315  snprintf(str1, BUFSIZE, "%sconfig/params.dtd", GetDataDir());
316  snprintf(str2, BUFSIZE, "<?xml-stylesheet type=\"text/xsl\" href=\"file:///%sconfig/style.xsl\"?>", GetDataDir());
317 
318  GfParmSetDTD (results, str1, str2);
319  GfParmCreateDirectory(0, results);
320  GfParmWriteFile(0, results, "Results");
321 }

324 void ReStoreRaceResults(const char *race)
325 {
326  int i;
327  int nCars;
328  tCarElt *car;
329  tSituation *s = ReInfo->s;
330  char *carName;
331  void *carparam;
332  void *results = ReInfo->results;
333  void *params = ReInfo->params;
334  const int BUFSIZE = 1024;
335  char buf[BUFSIZE], path[BUFSIZE], path2[BUFSIZE];
336 
337  /* Store the number of laps of the race */
338  switch (ReInfo->s->_raceType) {
339  case RM_TYPE_RACE:
340  car = s->cars[0];
341  if (car->_laps > s->_totLaps) car->_laps = s->_totLaps + 1;
342 
343  snprintf(path, BUFSIZE, "%s/%s/%s", ReInfo->track->name, RE_SECT_RESULTS, race);
344  GfParmListClean(results, path);
345  GfParmSetNum(results, path, RE_ATTR_LAPS, NULL, car->_laps - 1);
346 
347  ReApplyRaceTimePenalties();
348 
349  for (i = 0; i < s->_ncars; i++) {
350  snprintf(path, BUFSIZE, "%s/%s/%s/%s/%d", ReInfo->track->name, RE_SECT_RESULTS, race, RE_SECT_RANK, i + 1);
351  car = s->cars[i];
352  if (car->_laps > s->_totLaps) car->_laps = s->_totLaps + 1;
353 
354  GfParmSetStr(results, path, RE_ATTR_NAME, car->_name);
355 
356  snprintf(buf, BUFSIZE, "cars/%s/%s.xml", car->_carName, car->_carName);
357  carparam = GfParmReadFile(buf, GFPARM_RMODE_STD);
358  carName = GfParmGetName(carparam);
359 
360  GfParmSetStr(results, path, RE_ATTR_CAR, carName);
361  GfParmSetNum(results, path, RE_ATTR_INDEX, NULL, car->index);
362 
363  GfParmSetNum(results, path, RE_ATTR_LAPS, NULL, car->_laps - 1);
364  GfParmSetNum(results, path, RE_ATTR_TIME, NULL, car->_curTime + car->_penaltyTime);
365  GfParmSetNum(results, path, RE_ATTR_PENALTYTIME, NULL, car->_penaltyTime);
366  GfParmSetNum(results, path, RE_ATTR_BEST_LAP_TIME, NULL, car->_bestLapTime);
367  GfParmSetNum(results, path, RE_ATTR_TOP_SPEED, NULL, car->_topSpeed);
368  GfParmSetNum(results, path, RE_ATTR_DAMMAGES, NULL, car->_dammage);
369  GfParmSetNum(results, path, RE_ATTR_NB_PIT_STOPS, NULL, car->_nbPitStops);
370 
371  GfParmSetStr(results, path, RE_ATTR_MODULE, car->_modName);
372  GfParmSetNum(results, path, RE_ATTR_IDX, NULL, car->_driverIndex);
373 
374  snprintf(path2, BUFSIZE, "%s/%s/%d", race, RM_SECT_POINTS, i + 1);
375  GfParmSetNum(results, path, RE_ATTR_POINTS, NULL,
376  (int)GfParmGetNum(params, path2, RE_ATTR_POINTS, NULL, 0));
377 
378  GfParmReleaseHandle(carparam);
379  }
380  break;
381 
382  case RM_TYPE_PRACTICE:
383  car = s->cars[0];
384  snprintf(path, BUFSIZE, "%s/%s/%s", ReInfo->track->name, RE_SECT_RESULTS, race);
385  GfParmSetStr(results, path, RM_ATTR_DRVNAME, car->_name);
386  break;
387 
388  case RM_TYPE_QUALIF:
389  car = s->cars[0];
390  snprintf(path, BUFSIZE, "%s/%s/%s/%s", ReInfo->track->name, RE_SECT_RESULTS, race, RE_SECT_RANK);
391  nCars = GfParmGetEltNb(results, path);
392  for (i = nCars; i > 0; i--) {
393  snprintf(path, BUFSIZE, "%s/%s/%s/%s/%d", ReInfo->track->name, RE_SECT_RESULTS, race, RE_SECT_RANK, i);
394  float opponentBestLapTime = GfParmGetNum(results, path, RE_ATTR_BEST_LAP_TIME, NULL, 0);
395 
396  if (
397  (car->_bestLapTime != 0.0) &&
398  ((round(car->_bestLapTime*1000.0f) < round(opponentBestLapTime*1000.0f)) || (opponentBestLapTime == 0.0))
399  ) {
400  /* shift */
401  snprintf(path2, BUFSIZE, "%s/%s/%s/%s/%d", ReInfo->track->name, RE_SECT_RESULTS, race, RE_SECT_RANK, i + 1);
402  GfParmSetStr(results, path2, RE_ATTR_NAME, GfParmGetStr(results, path, RE_ATTR_NAME, ""));
403  GfParmSetStr(results, path2, RE_ATTR_CAR, GfParmGetStr(results, path, RE_ATTR_CAR, ""));
404  GfParmSetNum(results, path2, RE_ATTR_BEST_LAP_TIME, NULL, GfParmGetNum(results, path, RE_ATTR_BEST_LAP_TIME, NULL, 0));
405  GfParmSetStr(results, path2, RE_ATTR_MODULE, GfParmGetStr(results, path, RM_ATTR_MODULE, ""));
406  GfParmSetNum(results, path2, RE_ATTR_IDX, NULL, GfParmGetNum(results, path, RM_ATTR_IDX, NULL, 0));
407  snprintf(path, BUFSIZE, "%s/%s/%d", race, RM_SECT_POINTS, i + 1);
408  GfParmSetNum(results, path2, RE_ATTR_POINTS, NULL,
409  (int)GfParmGetNum(params, path, RE_ATTR_POINTS, NULL, 0));
410  } else {
411  break;
412  }
413  }
414  /* insert after */
415  snprintf(path, BUFSIZE, "%s/%s/%s/%s/%d", ReInfo->track->name, RE_SECT_RESULTS, race, RE_SECT_RANK, i + 1);
416  GfParmSetStr(results, path, RE_ATTR_NAME, car->_name);
417 
418  snprintf(buf, BUFSIZE, "cars/%s/%s.xml", car->_carName, car->_carName);
419  carparam = GfParmReadFile(buf, GFPARM_RMODE_STD);
420  carName = GfParmGetName(carparam);
421 
422  GfParmSetStr(results, path, RE_ATTR_CAR, carName);
423  GfParmSetNum(results, path, RE_ATTR_BEST_LAP_TIME, NULL, round(car->_bestLapTime*1000.0f)/1000.0f);
424  GfParmSetStr(results, path, RE_ATTR_MODULE, car->_modName);
425  GfParmSetNum(results, path, RE_ATTR_IDX, NULL, car->_driverIndex);
426  snprintf(path2, BUFSIZE, "%s/%s/%d", race, RM_SECT_POINTS, i + 1);
427  GfParmSetNum(results, path, RE_ATTR_POINTS, NULL,
428  (int)GfParmGetNum(params, path2, RE_ATTR_POINTS, NULL, 0));
429 
430  GfParmReleaseHandle(carparam);
431  break;
432  }
