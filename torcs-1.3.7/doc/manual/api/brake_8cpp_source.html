22 void SimBrakeConfig(void *hdle, const char *section, tBrake *brake)
23 {
24  tdble diam, area, mu;
25 
26  diam = GfParmGetNum(hdle, section, PRM_BRKDIAM, (char*)NULL, 0.2f);
27  area = GfParmGetNum(hdle, section, PRM_BRKAREA, (char*)NULL, 0.002f);
28  mu = GfParmGetNum(hdle, section, PRM_MU, (char*)NULL, 0.30f);
29  brake->coeff = diam * 0.5 * area * mu;
30 
31  brake->I = GfParmGetNum(hdle, section, PRM_INERTIA, (char*)NULL, 0.13f);
32  brake->radius = diam/2.0f;
33 }

36 void SimBrakeUpdate(tCar *car, tWheel *wheel, tBrake *brake)
37 {
38  brake->Tq = brake->coeff * brake->pressure;
39 
40  brake->temp -= fabs(car->DynGC.vel.x) * 0.0001 + 0.0002;
41  if (brake->temp < 0 ) brake->temp = 0;
42  brake->temp += brake->pressure * brake->radius * fabs(wheel->spinVel) * 0.00000000005;
43  if (brake->temp > 1.0) brake->temp = 1.0;
44 }
