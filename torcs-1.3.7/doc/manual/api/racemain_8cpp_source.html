134 void ReInitRules(tRmInfo* ReInfo)
135 {
136  // Invalidate best lap time when wall is hit?
137  const char* value = GfParmGetStr(ReInfo->params, ReInfo->_reRaceName, RM_ATTR_INVALIDATE_BEST_LAP_WALL_TOUCH, RM_VAL_YES);
138  if (!strcmp(value, RM_VAL_YES)) {
139  ReInfo->raceRules.enabled |= RmRaceRules::WALL_HIT_TIME_INVALIDATE;
140  }
141 
142  // Invalidate best lap time when corner is cut?
143  value = GfParmGetStr(ReInfo->params, ReInfo->_reRaceName, RM_ATTR_INVALIDATE_BEST_LAP_CORNER_CUT, RM_VAL_YES);
144  if (!strcmp(value, RM_VAL_YES)) {
145  ReInfo->raceRules.enabled |= RmRaceRules::CORNER_CUTTING_TIME_INVALIDATE;
146  }
147 
148  // Time penalty for corner cutting?
149  value = GfParmGetStr(ReInfo->params, ReInfo->_reRaceName, RM_ATTR_CORNER_CUT_TIME_PENALTY, RM_VAL_YES);
150  if (!strcmp(value, RM_VAL_YES)) {
151  ReInfo->raceRules.enabled |= RmRaceRules::CORNER_CUTTING_TIME_PENALTY;
152  }
153 
154  // Fuel consumption factor
155  tdble number = GfParmGetNum(ReInfo->params, ReInfo->_reRaceName, RM_ATTR_FUEL_FACTOR, NULL, 1.0f);
156  if (number < 0.0f) number = 0.0f; // Avoid negative factor
157  ReInfo->raceRules.fuelFactor = number;
158 
159  // Damage factor
160  number = GfParmGetNum(ReInfo->params, ReInfo->_reRaceName, RM_ATTR_DAMAGE_FACTOR, NULL, 1.0f);
161  if (number < 0.0f) number = 0.0f; // Avoid negative factor
162  ReInfo->raceRules.damageFactor = number;
163 
164  // Refuel fuel flow
165  number = GfParmGetNum(ReInfo->params, ReInfo->_reRaceName, RM_ATTR_REFUEL_FUEL_FLOW, NULL, 8.0f);
166  if (number < 1.0f) number = 1.0f; // Avoid division by zero or negative pit times
167  ReInfo->raceRules.refuelFuelFlow = number;
168 
169  // Damage repair factor
170  number = GfParmGetNum(ReInfo->params, ReInfo->_reRaceName, RM_ATTR_DAMAGE_REPAIR_FACTOR, NULL, 0.007f);
171  if (number < 0.0f) number = 0.0f; // Avoid negative pit times
172  ReInfo->raceRules.damageRepairFactor = number;
173 
174  // Pit stop base time (time for a stop even if nothing is done)
175  number = GfParmGetNum(ReInfo->params, ReInfo->_reRaceName, RM_ATTR_PITSTOP_BASE_TIME, NULL, 2.0f);
176  if (number < 0.0f) number = 0.0f; // Avoid negative pit times
177  ReInfo->raceRules.pitstopBaseTime = number;
178 
179  // Race specific pit speed limit, if available
180  number = ReInfo->track->pits.speedLimit;
181  number = GfParmGetNum(ReInfo->params, ReInfo->_reRaceName, RM_ATTR_PIT_SPEED_LIMIT, NULL, number);
182  ReInfo->track->pits.speedLimit = number;
183 }

186 int RePreRace(void)
187 {
188  tdble dist;
189  void *params = ReInfo->params;
190  void *results = ReInfo->results;
191  const int BUFSIZE = 1024;
192  char path[BUFSIZE];
193 
194  const char* raceName = ReInfo->_reRaceName = ReGetCurrentRaceName();
195  if (!raceName) {
196  return RM_QUIT;
197  }
198 
199  dist = GfParmGetNum(params, raceName, RM_ATTR_DISTANCE, NULL, 0);
200  if (dist < 0.001) {
201  ReInfo->s->_totLaps = (int)GfParmGetNum(params, raceName, RM_ATTR_LAPS, NULL, 30);
202  } else {
203  ReInfo->s->_totLaps = ((int)(dist / ReInfo->track->length)) + 1;
204  }
205  ReInfo->s->_maxDammage = (int)GfParmGetNum(params, raceName, RM_ATTR_MAX_DMG, NULL, 10000);
206 
207  const char* raceType = GfParmGetStr(params, raceName, RM_ATTR_TYPE, RM_VAL_RACE);
208  if (!strcmp(raceType, RM_VAL_RACE)) {
209  ReInfo->s->_raceType = RM_TYPE_RACE;
210  } else if (!strcmp(raceType, RM_VAL_QUALIF)) {
211  ReInfo->s->_raceType = RM_TYPE_QUALIF;
212  } else if (!strcmp(raceType, RM_VAL_PRACTICE)) {
213  ReInfo->s->_raceType = RM_TYPE_PRACTICE;
214  }
215 
216  ReInfo->s->_raceState = 0;
217 
218  /* Cleanup results */
219  snprintf(path, BUFSIZE, "%s/%s/%s", ReInfo->track->name, RE_SECT_RESULTS, raceName);
220  GfParmListClean(results, path);
221 
222  ReInitRules(ReInfo);
223 
224  return RM_SYNC | RM_NEXT_STEP;
225 }

229 static int reRaceRealStart(void)
230 {
231  int i, j;
232  int sw, sh, vw, vh;
233  tRobotItf *robot;
234  tReCarInfo *carInfo;
235  const int BUFSIZE = 1024;
236  char buf[BUFSIZE];
237  int foundHuman;
238  void *params = ReInfo->params;
239  void *results = ReInfo->results;
240  tSituation *s = ReInfo->s;
241 
242  RmLoadingScreenSetText("Loading Simulation Engine...");
243  const char* dllname = GfParmGetStr(ReInfo->_reParam, "Modules", "simu", "");
244  snprintf(buf, BUFSIZE, "%smodules/simu/%s.%s", GetLibDir (), dllname, DLLEXT);
245  if (GfModLoad(0, buf, &ReRaceModList)) return RM_QUIT;
246  ReRaceModList->modInfo->fctInit(ReRaceModList->modInfo->index, &ReInfo->_reSimItf);
247 
248  if (ReInitCars()) {
249  return RM_QUIT;
250  }
251 
252  /* Blind mode or not */
253  if (ReInfo->_displayMode != RM_DISP_MODE_CONSOLE) {
254  ReInfo->_displayMode = RM_DISP_MODE_NORMAL;
255  ReInfo->_reGameScreen = ReScreenInit();
256  foundHuman = 0;
257  for (i = 0; i < s->_ncars; i++) {
258  if (s->cars[i]->_driverType == RM_DRV_HUMAN) {
259  foundHuman = 1;
260  break;
261  }
262  }
263  if (!foundHuman) {
264  if (!strcmp(GfParmGetStr(params, ReInfo->_reRaceName, RM_ATTR_DISPMODE, RM_VAL_VISIBLE), RM_VAL_INVISIBLE)) {
265  ReInfo->_displayMode = RM_DISP_MODE_NONE;
266  ReInfo->_reGameScreen = ReResScreenInit();
267  }
268  }
269  }
270 
271  if (!(ReInfo->s->_raceType == RM_TYPE_QUALIF) ||
272  ((int)GfParmGetNum(results, RE_SECT_CURRENT, RE_ATTR_CUR_DRIVER, NULL, 1) == 1))
273  {
274  RmLoadingScreenStart(ReInfo->_reName, "data/img/splash-qrloading.png");
275  }
276 
277  for (i = 0; i < s->_ncars; i++) {
278  snprintf(buf, BUFSIZE, "Initializing Driver %s...", s->cars[i]->_name);
279  RmLoadingScreenSetText(buf);
280  robot = s->cars[i]->robot;
281  robot->rbNewRace(robot->index, s->cars[i], s);
282  }
283  carInfo = ReInfo->_reCarInfo;
284 
285  ReInfo->_reSimItf.update(s, RCM_MAX_DT_SIMU, -1);
286  for (i = 0; i < s->_ncars; i++) {
287  carInfo[i].prevTrkPos = s->cars[i]->_trkPos;
288  }
289 
290  RmLoadingScreenSetText("Running Prestart...");
291  for (i = 0; i < s->_ncars; i++) {
292  memset(&(s->cars[i]->ctrl), 0, sizeof(tCarCtrl));
293  s->cars[i]->ctrl.brakeCmd = 1.0;
294  }
295  for (j = 0; j < ((int)(1.0 / RCM_MAX_DT_SIMU)); j++) {
296  ReInfo->_reSimItf.update(s, RCM_MAX_DT_SIMU, -1);
297  }
298 
299  if (ReInfo->_displayMode == RM_DISP_MODE_NONE) {
300  if (ReInfo->s->_raceType == RM_TYPE_QUALIF) {
301  ReUpdateQualifCurRes(s->cars[0]);
302  } else {
303  snprintf(buf, BUFSIZE, "%s on %s", s->cars[0]->_name, ReInfo->track->name);
304  ReResScreenSetTitle(buf);
305  }
306  }
307 
308  RmLoadingScreenSetText("Ready.");
309 
310  ReInfo->_reTimeMult = 1.0;
311  ReInfo->_reLastTime = -1.0;
312  ReInfo->s->currentTime = -2.0;
313  ReInfo->s->deltaTime = RCM_MAX_DT_SIMU;
314 
315  ReInfo->s->_raceState = RM_RACE_STARTING;
316 
317  if ((ReInfo->_displayMode != RM_DISP_MODE_CONSOLE) && ReInfo->_reGraphicItf.initview != 0) {
318  GfScrGetSize(&sw, &sh, &vw, &vh);
319  ReInfo->_reGraphicItf.initview((sw-vw)/2, (sh-vh)/2, vw, vh, GR_VIEW_STD, ReInfo->_reGameScreen);
320 
321  if (ReInfo->_displayMode == RM_DISP_MODE_NORMAL) {
322  /* RmLoadingScreenSetText("Loading Cars 3D Objects..."); */
323  stopMenuMusic();
324  ReInfo->_reGraphicItf.initcars(s);
325  }
326 
327  GfuiScreenActivate(ReInfo->_reGameScreen);
328  }
329 
330  return RM_SYNC | RM_NEXT_STEP;
331 }

357 /* return state mode */
358 int ReRaceStart(void)
359 {
360  int i;
361  int nCars;
362  int maxCars;
363  const char *raceName = ReInfo->_reRaceName;
364  void *params = ReInfo->params;
365  void *results = ReInfo->results;
366  const int BUFSIZE = 1024;
367  char path[BUFSIZE], path2[BUFSIZE];
368 
369  FREEZ(ReInfo->_reCarInfo);
370  ReInfo->_reCarInfo = (tReCarInfo*)calloc(GfParmGetEltNb(params, RM_SECT_DRIVERS), sizeof(tReCarInfo));
371 
372  /* Drivers starting order */
373  GfParmListClean(params, RM_SECT_DRIVERS_RACING);
374  if (ReInfo->s->_raceType == RM_TYPE_QUALIF) {
375  i = (int)GfParmGetNum(results, RE_SECT_CURRENT, RE_ATTR_CUR_DRIVER, NULL, 1);
376  if (i == 1) {
377  RmLoadingScreenStart(ReInfo->_reName, "data/img/splash-qrloading.png");
378  RmLoadingScreenSetText("Preparing Starting Grid...");
379  } else {
380  RmShutdownLoadingScreen();
381  }
382 
383  snprintf(path, BUFSIZE, "%s/%d", RM_SECT_DRIVERS, i);
384  snprintf(path2, BUFSIZE, "%s/%d", RM_SECT_DRIVERS_RACING, 1);
385  GfParmSetStr(params, path2, RM_ATTR_MODULE, GfParmGetStr(params, path, RM_ATTR_MODULE, ""));
386  GfParmSetNum(params, path2, RM_ATTR_IDX, NULL, GfParmGetNum(params, path, RM_ATTR_IDX, NULL, 0));
387  } else {
388  RmLoadingScreenStart(ReInfo->_reName, "data/img/splash-qrloading.png");
389  RmLoadingScreenSetText("Preparing Starting Grid...");
390 
391  const char* gridType = GfParmGetStr(params, raceName, RM_ATTR_START_ORDER, RM_VAL_DRV_LIST_ORDER);
392  if (!strcmp(gridType, RM_VAL_LAST_RACE_ORDER)) {
393  /* Starting grid in the arrival of the previous race */
394  nCars = GfParmGetEltNb(params, RM_SECT_DRIVERS);
395  maxCars = (int)GfParmGetNum(params, raceName, RM_ATTR_MAX_DRV, NULL, 100);
396  nCars = MIN(nCars, maxCars);
397  const char* prevRaceName = ReGetPrevRaceName();
398  if (!prevRaceName) {
399  return RM_QUIT;
400  }
401  for (i = 1; i < nCars + 1; i++) {
402  snprintf(path, BUFSIZE, "%s/%s/%s/%s/%d", ReInfo->track->name, RE_SECT_RESULTS, prevRaceName, RE_SECT_RANK, i);
403  snprintf(path2, BUFSIZE, "%s/%d", RM_SECT_DRIVERS_RACING, i);
404  GfParmSetStr(params, path2, RM_ATTR_MODULE, GfParmGetStr(results, path, RE_ATTR_MODULE, ""));
405  GfParmSetNum(params, path2, RM_ATTR_IDX, NULL, GfParmGetNum(results, path, RE_ATTR_IDX, NULL, 0));
406  }
407  } else if (!strcmp(gridType, RM_VAL_LAST_RACE_RORDER)) {
408  /* Starting grid in the reversed arrival order of the previous race */
409  nCars = GfParmGetEltNb(params, RM_SECT_DRIVERS);
410  maxCars = (int)GfParmGetNum(params, raceName, RM_ATTR_MAX_DRV, NULL, 100);
411  nCars = MIN(nCars, maxCars);
412  const char* prevRaceName = ReGetPrevRaceName();
413  if (!prevRaceName) {
414  return RM_QUIT;
415  }
416  for (i = 1; i < nCars + 1; i++) {
417  snprintf(path, BUFSIZE, "%s/%s/%s/%s/%d", ReInfo->track->name, RE_SECT_RESULTS, prevRaceName, RE_SECT_RANK, nCars - i + 1);
418  snprintf(path2, BUFSIZE, "%s/%d", RM_SECT_DRIVERS_RACING, i);
419  GfParmSetStr(params, path2, RM_ATTR_MODULE, GfParmGetStr(results, path, RE_ATTR_MODULE, ""));
420  GfParmSetNum(params, path2, RM_ATTR_IDX, NULL, GfParmGetNum(results, path, RE_ATTR_IDX, NULL, 0));
421  }
422  } else {
423  /* Starting grid in the drivers list order */
424  nCars = GfParmGetEltNb(params, RM_SECT_DRIVERS);
425  maxCars = (int)GfParmGetNum(params, raceName, RM_ATTR_MAX_DRV, NULL, 100);
426  nCars = MIN(nCars, maxCars);
427  for (i = 1; i < nCars + 1; i++) {
428  snprintf(path, BUFSIZE, "%s/%d", RM_SECT_DRIVERS, i);
429  snprintf(path2, BUFSIZE, "%s/%d", RM_SECT_DRIVERS_RACING, i);
430  GfParmSetStr(params, path2, RM_ATTR_MODULE, GfParmGetStr(params, path, RM_ATTR_MODULE, ""));
431  GfParmSetNum(params, path2, RM_ATTR_IDX, NULL, GfParmGetNum(params, path, RM_ATTR_IDX, NULL, 0));
432  }
433  }
434  }
435 
436  if (ReInfo->_displayMode != RM_DISP_MODE_CONSOLE) {
437  if (!strcmp(GfParmGetStr(params, ReInfo->_reRaceName, RM_ATTR_SPLASH_MENU, RM_VAL_NO), RM_VAL_YES)) {
438  RmShutdownLoadingScreen();
439  RmDisplayStartRace(ReInfo, StartRaceHookInit(), AbandonRaceHookInit());
440  return RM_ASYNC | RM_NEXT_STEP;
441  }
442  }
443 
444  return reRaceRealStart();
445 }

519 int ReRaceStop(void)
520 {
521  void *params = ReInfo->params;
522  ReInfo->_reGraphicItf.muteformenu();
523 
524  if (!strcmp(GfParmGetStr(params, ReInfo->_reRaceName, RM_ATTR_ALLOW_RESTART, RM_VAL_NO), RM_VAL_NO)) {
525  StopScrHandle = RmTriStateScreen("Race Stopped",
526  "Abandon Race", "Abort current race", AbortRaceHookInit(),
527  "Resume Race", "Return to Race", BackToRaceHookInit(),
528  "Quit Game", "Quit the game", QuitHookInit());
529  } else {
530  if (
531  (ReInfo->s->raceInfo.type == RM_TYPE_PRACTICE || ReInfo->s->raceInfo.type == RM_TYPE_QUALIF) &&
532  (ReInfo->s->raceInfo.ncars == 1) &&
533  (ReInfo->carList[0].info.driverType == RM_DRV_HUMAN)
534  ) {
535  tCarElt* carElt = &ReInfo->carList[0];
536  static const char* label[5] = { "Restart Race", "Setup Car, Restart", "Abandon Race", "Resume Race", "Quit Game" };
537  static const char* tip[5] = { "Restart the current race", "Setup car and restart the current race", "Abort the current race", "Return to the race", "Quit TORCS" };
538  void* screen[5];
539 
540  screen[0] = RestartRaceHookInit();
541  screen[1] = RmCarSetupScreenInit(RestartRaceHookInit(), carElt, ReInfo);
542  screen[2] = AbortRaceHookInit();
543  screen[3] = BackToRaceHookInit();
544  screen[4] = QuitHookInit();
545 
546  StopScrHandle = RmNStateScreen("Race Stopped", label, tip, screen, 5);
547  } else {
548  StopScrHandle = RmFourStateScreen("Race Stopped",
549  "Restart Race", "Restart the current race", RestartRaceHookInit(),
550  "Abandon Race", "Abort current race", AbortRaceHookInit(),
551  "Resume Race", "Return to Race", BackToRaceHookInit(),
552  "Quit Game", "Quit the game", QuitHookInit());
553  }
554  }
555  return RM_ASYNC | RM_NEXT_STEP;
556 }

562  void *params = ReInfo->params;
563  void *results = ReInfo->results;
564 
565  ReRaceCleanup();
566 
567  if (ReInfo->s->_raceType == RM_TYPE_QUALIF) {

569  curDrvIdx++;
570  if (curDrvIdx > GfParmGetEltNb(params, RM_SECT_DRIVERS)) {

575  return RM_SYNC | RM_NEXT_RACE;
