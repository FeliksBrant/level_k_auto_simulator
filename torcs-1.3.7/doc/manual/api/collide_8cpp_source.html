25 void SimCarCollideZ(tCar *car)
26 {
27  int i;
28  t3Dd normal;
29  tdble dotProd;
30  tWheel *wheel;
31  const float CRASH_THRESHOLD = -5.0f;
32 
33  if (car->carElt->_state & RM_CAR_STATE_NO_SIMU) {
34  return;
35  }
36 
37  for (i = 0; i < 4; i++) {
38  wheel = &(car->wheel[i]);
39  if (wheel->state & SIM_SUSP_COMP) {
40  car->DynGCg.pos.z += wheel->susp.spring.packers - wheel->rideHeight;
41  RtTrackSurfaceNormalL(&(wheel->trkPos), &normal);
42  dotProd = (car->DynGCg.vel.x * normal.x + car->DynGCg.vel.y * normal.y + car->DynGCg.vel.z * normal.z) * wheel->trkPos.seg->surface->kRebound;
43  if (dotProd < 0.0f) {
44  if (dotProd < CRASH_THRESHOLD) {
45  car->collision |= SEM_COLLISION_Z_CRASH;
46  }
47  car->collision |= SEM_COLLISION | SEM_COLLISION_Z;
48  car->DynGCg.vel.x -= normal.x * dotProd;
49  car->DynGCg.vel.y -= normal.y * dotProd;
50  car->DynGCg.vel.z -= normal.z * dotProd;
51  if ((car->carElt->_state & RM_CAR_STATE_FINISH) == 0) {
52  car->dammage += (int)(wheel->trkPos.seg->surface->kDammage * fabs(dotProd) * rulesDamageFactor * simDammageFactor[car->carElt->_skillLevel]);
53  }
54  }
55  }
56  }

61 // Collision of car/track borders.
62 // Be aware that it does not work for convex edges (e.g. e-track-2, end of the straight, left),
63 // because under these conditions it is possible that all car corners are on the track and
64 // the car body is partially outside the track.
65 void SimCarCollideXYScene(tCar *car)
66 {
67  tTrackSeg *seg = car->trkPos.seg;
68  tTrkLocPos trkpos;
69  int i;
70  tDynPt *corner;
71  tdble initDotProd;
72  tdble dotProd, cx, cy, dotprod2;
73  tTrackBarrier *curBarrier;
74  tdble dmg;
75 
76  if (car->carElt->_state & RM_CAR_STATE_NO_SIMU) {
77  return;
78  }
79 
80  corner = &(car->corner[0]);
81  for (i = 0; i < 4; i++, corner++) {
82  seg = car->trkPos.seg;
83  RtTrackGlobal2Local(seg, corner->pos.ax, corner->pos.ay, &trkpos, TR_LPOS_TRACK);
84  seg = trkpos.seg;
85  tdble toSide;
86 
87  if (trkpos.toRight < 0.0) {
88  // collision with right border.
89  curBarrier = seg->barrier[TR_SIDE_RGT];
90  toSide = trkpos.toRight;
91  } else if (trkpos.toLeft < 0.0) {
92  // collision with left border.
93  curBarrier = seg->barrier[TR_SIDE_LFT];
94  toSide = trkpos.toLeft;
95  } else {
96  continue;
97  }
98 
99  const tdble& nx = curBarrier->normal.x;
100  const tdble& ny = curBarrier->normal.y;
101 
102  car->DynGCg.pos.x -= nx * toSide;
103  car->DynGCg.pos.y -= ny * toSide;
104 
105  // Corner position relative to center of gravity.
106  cx = corner->pos.ax - car->DynGCg.pos.x;
107  cy = corner->pos.ay - car->DynGCg.pos.y;
108 
109  car->blocked = 1;
110  car->collision |= SEM_COLLISION;
111 
112  // Impact speed perpendicular to barrier (of corner).
113  initDotProd = nx * corner->vel.x + ny * corner->vel.y;
114 
115  // Compute dmgDotProd (base value for later damage) with a heuristic.
116  tdble absvel = MAX(1.0, sqrt(car->DynGCg.vel.x*car->DynGCg.vel.x + car->DynGCg.vel.y*car->DynGCg.vel.y));
117  tdble GCgnormvel = car->DynGCg.vel.x*nx + car->DynGCg.vel.y*ny;
118  tdble cosa = GCgnormvel/absvel;
119  tdble dmgDotProd = GCgnormvel*cosa;
120 
121  dotProd = initDotProd * curBarrier->surface->kFriction;
122  car->DynGCg.vel.x -= nx * dotProd;
123  car->DynGCg.vel.y -= ny * dotProd;
124  dotprod2 = (nx * cx + ny * cy);
125 
126  // Angular velocity change caused by friction of colliding car part with wall.
127  static tdble VELSCALE = 10.0f;
128  static tdble VELMAX = 6.0f;
129  car->DynGCg.vel.az -= dotprod2 * dotProd / VELSCALE;
130  if (fabs(car->DynGCg.vel.az) > VELMAX) {
131  car->DynGCg.vel.az = SIGN(car->DynGCg.vel.az) * VELMAX;
132  }
133 
134  // Damage.
135  dotProd = initDotProd;
136  if (dotProd < 0.0f && (car->carElt->_state & RM_CAR_STATE_FINISH) == 0) {
137  dmg = curBarrier->surface->kDammage * (0.5f*dmgDotProd*dmgDotProd + 0.005f*fabs(1.0f-cosa)*absvel) * rulesDamageFactor * simDammageFactor[car->carElt->_skillLevel];
138  car->dammage += (int) dmg;
139  } else {
140  dmg = 0.0f;
141  }
142 
143  dotProd *= curBarrier->surface->kRebound;
144 
145  // If the car moves toward the barrier, rebound.
146  if (dotProd < 0.0f) {
147  car->collision |= SEM_COLLISION_XYSCENE;
148  car->normal.x = nx * dmg;
149  car->normal.y = ny * dmg;
150  car->collpos.x = corner->pos.ax;
151  car->collpos.y = corner->pos.ay;
152  car->DynGCg.vel.x -= nx * dotProd;
153  car->DynGCg.vel.y -= ny * dotProd;
154  }
155  }
156 }

159 static void SimCarCollideResponse(void * /*dummy*/, DtObjectRef obj1, DtObjectRef obj2, const DtCollData *collData)
160 {
161  sgVec2 n; // Collision normal delivered by solid: Global(point1) - Global(point2)
162  tCar *car[2]; // The cars.
163  sgVec2 p[2]; // Collision points delivered by solid, in body local coordinates.
164  sgVec2 r[2]; // Collision point relative to center of gravity.
165  sgVec2 vp[2]; // Speed of collision point in world coordinate system.
166  sgVec3 pt[2]; // Collision points in global coordinates.
167 
168  int i;
169 
170  car[0] = (tCar*)obj1;
171  car[1] = (tCar*)obj2;
172 
173  // Handle cars collisions during pit stops as well.
174  static const int NO_SIMU_WITHOUT_PIT = RM_CAR_STATE_NO_SIMU & ~RM_CAR_STATE_PIT;
175 
176  if ((car[0]->carElt->_state & NO_SIMU_WITHOUT_PIT) ||
177  (car[1]->carElt->_state & NO_SIMU_WITHOUT_PIT))
178  {
179  return;
180  }
181 
182  if (car[0]->carElt->index < car[1]->carElt->index) {
183  // vector conversion from double to float.
184  p[0][0] = (float)collData->point1[0];
185  p[0][1] = (float)collData->point1[1];
186  p[1][0] = (float)collData->point2[0];
187  p[1][1] = (float)collData->point2[1];
188  n[0] = (float)collData->normal[0];
189  n[1] = (float)collData->normal[1];
190  } else {
191  // swap the cars (not the same for the simu).
192  car[0] = (tCar*)obj2;
193  car[1] = (tCar*)obj1;
194  p[0][0] = (float)collData->point2[0];
195  p[0][1] = (float)collData->point2[1];
196  p[1][0] = (float)collData->point1[0];
197  p[1][1] = (float)collData->point1[1];
198  n[0] = -(float)collData->normal[0];
199  n[1] = -(float)collData->normal[1];
200  }
201 
202  sgNormaliseVec2(n);
203 
204  sgVec2 rg[2]; // radius oriented in global coordinates, still relative to CG (rotated aroung CG).
205  tCarElt *carElt;
206 
207  for (i = 0; i < 2; i++) {
208  // vector GP (Center of gravity to collision point). p1 and p2 are delivered from solid as
209  // points in the car coordinate system.
210  sgSubVec2(r[i], p[i], (const float*)&(car[i]->statGC));
211 
212  // Speed of collision points, linear motion of center of gravity (CG) plus rotational
213  // motion around the CG.
214  carElt = car[i]->carElt;
215  float sina = sin(carElt->_yaw);
216  float cosa = cos(carElt->_yaw);
217  rg[i][0] = r[i][0]*cosa - r[i][1]*sina;
218  rg[i][1] = r[i][0]*sina + r[i][1]*cosa;
219 
220  vp[i][0] = car[i]->DynGCg.vel.x - car[i]->DynGCg.vel.az * rg[i][1];
221  vp[i][1] = car[i]->DynGCg.vel.y + car[i]->DynGCg.vel.az * rg[i][0];
222  }
223 
224  // Relative speed of collision points.
225  sgVec2 v1ab;
226  sgSubVec2(v1ab, vp[0], vp[1]);
227 
228  // try to separate the cars. The computation is necessary because dtProceed is not called till
229  // the collision is resolved.
230  for (i = 0; i < 2; i++) {
231  sgCopyVec2(pt[i], r[i]);
232  pt[i][2] = 0.0f;
233  // Transform points relative to cars local coordinate system into global coordinates.
234  sgFullXformPnt3(pt[i], car[i]->carElt->_posMat);
235  }
236 
237  // Compute distance of collision points.
238  sgVec3 pab;
239  sgSubVec2(pab, pt[1], pt[0]);
240  float distpab = sgLengthVec2(pab);
241 
242  sgVec2 tmpv;
243 
244  sgScaleVec2(tmpv, n, MIN(distpab, 0.05));
245  // No "for" loop here because of subtle difference AddVec/SubVec.
246  if (car[0]->blocked == 0 && !(car[0]->carElt->_state & RM_CAR_STATE_NO_SIMU)) {
247  sgAddVec2((float*)&(car[0]->DynGCg.pos), tmpv);
248  car[0]->blocked = 1;
249  }
250  if (car[1]->blocked == 0 && !(car[1]->carElt->_state & RM_CAR_STATE_NO_SIMU)) {
251  sgSubVec2((float*)&(car[1]->DynGCg.pos), tmpv);
252  car[1]->blocked = 1;
253  }
254 
255  // Doing no dammage and correction if the cars are moving out of each other.
256  if (sgScalarProductVec2(v1ab, n) > 0) {
257  return;
258  }
259 
260  // impulse.
261  float rpn[2];
262  rpn[0] = sgScalarProductVec2(rg[0], n);
263  rpn[1] = sgScalarProductVec2(rg[1], n);
264 
265  // Pseudo cross product to find out if we are left or right.
266  // TODO: SIGN, scrap value?
267  float rpsign[2];
268  rpsign[0] = n[0]*rg[0][1] - n[1]*rg[0][0];
269  rpsign[1] = -n[0]*rg[1][1] + n[1]*rg[1][0];
270 
271  const float e = 1.0f; // energy restitution
272 
273  float j = -(1.0f + e) * sgScalarProductVec2(v1ab, n) /
274  ((car[0]->Minv + car[1]->Minv) +
275  rpn[0] * rpn[0] * car[0]->Iinv.z + rpn[1] * rpn[1] * car[1]->Iinv.z);
276 
277  for (i = 0; i < 2; i++) {
278  if (car[i]->carElt->_state & RM_CAR_STATE_NO_SIMU) {
279  continue;
280  }
281 
282  // Damage.
283  tdble damFactor, atmp;
284  atmp = atan2(r[i][1], r[i][0]);
285  if (fabs(atmp) < (PI / 3.0)) {
286  // Front collision gives more damage.
287  damFactor = 1.5f;
288  } else {
289  // Rear collision gives less damage.
290  damFactor = 1.0f;
291  }
292 
293  if ((car[i]->carElt->_state & RM_CAR_STATE_FINISH) == 0) {
294  car[i]->dammage += (int)(CAR_DAMMAGE * fabs(j) * damFactor * rulesDamageFactor * simDammageFactor[car[i]->carElt->_skillLevel]);
295  }
296 
297  // Compute collision velocity.
298  const float ROT_K = 1.0f;
299 
300  float js = (i == 0) ? j : -j;
301  sgScaleVec2(tmpv, n, js * car[i]->Minv);
302  sgVec2 v2a;
303 
304  if (car[i]->collision & SEM_COLLISION_CAR) {
305  sgAddVec2(v2a, (const float*)&(car[i]->VelColl.x), tmpv);
306  car[i]->VelColl.az = car[i]->VelColl.az + js * rpsign[i] * rpn[i] * car[i]->Iinv.z * ROT_K;
307  } else {
308  sgAddVec2(v2a, (const float*)&(car[i]->DynGCg.vel), tmpv);
309  car[i]->VelColl.az = car[i]->DynGCg.vel.az + js * rpsign[i] * rpn[i] * car[i]->Iinv.z * ROT_K;
310  }
311 
312  static float VELMAX = 3.0f;
313  if (fabs(car[i]->VelColl.az) > VELMAX) {
314  car[i]->VelColl.az = SIGN(car[i]->VelColl.az) * VELMAX;
315  }
316 
317  sgCopyVec2((float*)&(car[i]->VelColl.x), v2a);
318 
319  // Move the car for the collision lib.
320  tCarElt *carElt = car[i]->carElt;
321  sgMakeCoordMat4(carElt->pub.posMat, car[i]->DynGCg.pos.x - carElt->_statGC_x, car[i]->DynGCg.pos.y - carElt->_statGC_y,
322  car[i]->DynGCg.pos.z - carElt->_statGC_z, RAD2DEG(carElt->_yaw),
323  RAD2DEG(carElt->_roll), RAD2DEG(carElt->_pitch));
324  dtSelectObject(car[i]);
325  dtLoadIdentity();
326  dtMultMatrixf((const float *)(carElt->_posMat));
327 
328  car[i]->collision |= SEM_COLLISION_CAR;
329  }
330 }

333 // Collision response for walls.
334 // TODO: Separate common code with car-car collision response.
335 static void SimCarWallCollideResponse(void *clientdata, DtObjectRef obj1, DtObjectRef obj2, const DtCollData *collData)
336 {
337  tCar* car; // The car colliding with the wall.
338  float nsign; // Normal direction correction for collision plane.
339  sgVec2 p; // Cars collision point delivered by solid.
340 
341  // TODO: If other movable objects are added which could collide with the wall, it will be
342  // necessary to validate if the object is actually a car.
343 
344  if (obj1 == clientdata) {
345  car = (tCar*) obj2;
346  nsign = -1.0f;
347  p[0] = (float) collData->point2[0];
348  p[1] = (float) collData->point2[1];
349  } else {
350  car = (tCar*) obj1;
351  nsign = 1.0f;
352  p[0] = (float) collData->point1[0];
353  p[1] = (float) collData->point1[1];
354  }
355 
356  sgVec2 n; // Collision normal delivered by solid, corrected such that it points away from the wall.
357  n[0] = nsign * (float) collData->normal[0];
358  n[1] = nsign * (float) collData->normal[1];
359  float pdist = sgLengthVec2(n); // Distance of collision points.
360  sgNormaliseVec2(n);
361 
362  sgVec2 r;
363  sgSubVec2(r, p, (const float*)&(car->statGC));
364 
365  tCarElt *carElt = car->carElt;
366 
367  sgVec2 vp; // Speed of car collision point in global frame of reference.
368  sgVec2 rg; // raduis oriented in global coordinates, still relative to CG (rotated aroung CG).
369 
370  float sina = sin(carElt->_yaw);
371  float cosa = cos(carElt->_yaw);
372  rg[0] = r[0]*cosa - r[1]*sina;
373  rg[1] = r[0]*sina + r[1]*cosa;
374 
375  vp[0] = car->DynGCg.vel.x - car->DynGCg.vel.az * rg[1];
376  vp[1] = car->DynGCg.vel.y + car->DynGCg.vel.az * rg[0];
377 
378  sgVec2 tmpv;
379  static const float CAR_MIN_MOVEMENT = 0.02f;
380  static const float CAR_MAX_MOVEMENT = 0.05f;
381  sgScaleVec2(tmpv, n, MIN(MAX(pdist, CAR_MIN_MOVEMENT), CAR_MAX_MOVEMENT));
382  if (car->blocked == 0) {
383  sgAddVec2((float*)&(car->DynGCg.pos), tmpv);
384  car->blocked = 1;
385  }
386 
387  // Doing no dammage and correction if the cars are moving out of each other.
388  if (sgScalarProductVec2(vp, n) > 0) {
389  return;
390  }
391 
392  float rp = sgScalarProductVec2(rg, n);
393 
394  // Pesudo cross product to find out if we are left or right.
395  // TODO: SIGN, scrap value?
396  float rpsign = n[0]*rg[1] - n[1]*rg[0];
397 
398  const float e = 1.0f; // energy restitution
399  float j = -(1.0f + e) * sgScalarProductVec2(vp, n) / (car->Minv + rp * rp * car->Iinv.z);
400  const float ROT_K = 0.5f;
401 
402  // Damage.
403  tdble damFactor, atmp;
404  atmp = atan2(r[1], r[0]);
405  if (fabs(atmp) < (PI / 3.0)) {
406  // Front collision gives more damage.
407  damFactor = 1.5f;
408  } else {
409  // Rear collision gives less damage.
410  damFactor = 1.0f;
411  }
412 
413  static const float DMGFACTOR = 0.00002f;
414  if ((car->carElt->_state & RM_CAR_STATE_FINISH) == 0) {
415  car->dammage += (int)(CAR_DAMMAGE * (DMGFACTOR*j*j) * damFactor * rulesDamageFactor * simDammageFactor[car->carElt->_skillLevel]);
416  }
417 
418  sgScaleVec2(tmpv, n, j * car->Minv);
419  sgVec2 v2a;
420 
421  if (car->collision & SEM_COLLISION_CAR) {
422  sgAddVec2(v2a, (const float*)&(car->VelColl.x), tmpv);
423  car->VelColl.az = car->VelColl.az + j * rp * rpsign * car->Iinv.z * ROT_K;
424  } else {
425  sgAddVec2(v2a, (const float*)&(car->DynGCg.vel), tmpv);
426  car->VelColl.az = car->DynGCg.vel.az + j * rp * rpsign * car->Iinv.z * ROT_K;
427  }
428 
429  static float VELMAX = 3.0f;
430  if (fabs(car->VelColl.az) > VELMAX) {
431  car->VelColl.az = SIGN(car->VelColl.az) * VELMAX;
432  }
433 
434  sgCopyVec2((float*)&(car->VelColl.x), v2a);
435 
436  // Move the car for the collision lib.
437  sgMakeCoordMat4(carElt->pub.posMat, car->DynGCg.pos.x - carElt->_statGC_x, car->DynGCg.pos.y - carElt->_statGC_y,
438  car->DynGCg.pos.z - carElt->_statGC_z, RAD2DEG(carElt->_yaw),
439  RAD2DEG(carElt->_roll), RAD2DEG(carElt->_pitch));
440  dtSelectObject(car);
441  dtLoadIdentity();
442  dtMultMatrixf((const float *)(carElt->_posMat));
443 
444  car->collision |= SEM_COLLISION_CAR;
445 }

500 // Searching backwards for the first non wall segments to start.
501 // Used for the creation of collision detection model of the wall.
502 static tTrackSeg *getFirstWallStart(tTrackSeg *start, int side)
503 {
504  tTrackSeg *first = start;
505 
506  // Moving backward out of wall.
507  do {
508  // A wall is a wall on the track if it is not the barrier at the same time.
509  if (first->side[side] != NULL && first->side[side]->style == TR_WALL &&
510  first->side[side]->side[side] != NULL)
511  {
512  first = first->prev;
513  } else {
514  break;
515  }
516  } while (first != start);
517 
518  // Searching forward for the first wall segment.
519  start = first;
520  do {
521  if (first->side[side] != NULL && first->side[side]->style == TR_WALL &&
522  first->side[side]->side[side] != NULL)
523  {
524  return first;
525  } else {
526  first = first->next;
527  }
528  } while (first != start);
529 
530  return NULL;
531 }

534 // Create the left walls, start must point to a segment with a wall on the left and a leading
535 // non wall segment.
536 // FIXME: Does not work for a closed ring "by design".
537 void buildWalls(tTrackSeg *start, int side) {
538 
539  if (start == NULL) {
540  return;
541  }
542 
543  tTrackSeg *current = start;
544  bool close = false;
545 
546  // Going down the wall.
547  do {
548  tTrackSeg* s = current->side[side];
549  tTrackSeg* p = current->prev->side[side];
550  tTrackSeg* n = current->next->side[side];
551 
552  // Current segment is a wall?
553  if (s != NULL && s->style == TR_WALL && s->side[side] != NULL) {
554 
555  float h = s->height;
556  t3Dd svl = s->vertex[TR_SL];
557  t3Dd svr = s->vertex[TR_SR];
558  t3Dd evl = s->vertex[TR_EL];
559  t3Dd evr = s->vertex[TR_ER];

644  }
