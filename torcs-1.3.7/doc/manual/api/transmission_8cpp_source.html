25 void SimTransmissionConfig(tCar *car)
26 {
27  void *hdle = car->params;
28  tCarElt *carElt = car->carElt;
29  //tdble clutchI;
30  tTransmission *trans = &(car->transmission);
31  tClutch *clutch = &(trans->clutch);
32  tDifferential *differential;
33  int i, j;
34  tdble gRatio, fRatio, gEff; //fEff;
35  tdble gearI;
36  const int BUFSIZE = 256;
37  char path[BUFSIZE];
38 
39  //clutchI = GfParmGetNum(hdle, SECT_CLUTCH, PRM_INERTIA, (char*)NULL, 0.12f);
40  const char* transType = GfParmGetStr(hdle, SECT_DRIVETRAIN, PRM_TYPE, VAL_TRANS_RWD);
41  clutch->releaseTime = GfParmGetNum(hdle, SECT_GEARBOX, PRM_SHIFTTIME, (char*)NULL, 0.2f);
42 
43  fRatio = 0;
44  gEff = 0;
45 
46  /* Link between the differentials */
47  for (j = 0; j < 2; j++) {
48  trans->differential[TRANS_FRONT_DIFF].inAxis[j] = &(car->wheel[j].feedBack);
49  trans->differential[TRANS_FRONT_DIFF].outAxis[j] = &(car->wheel[j].in);
50  }
51 
52  for (j = 0; j < 2; j++) {
53  trans->differential[TRANS_REAR_DIFF].inAxis[j] = &(car->wheel[2+j].feedBack);
54  trans->differential[TRANS_REAR_DIFF].outAxis[j] = &(car->wheel[2+j].in);
55  }
56 
57  trans->differential[TRANS_CENTRAL_DIFF].inAxis[0] = &(trans->differential[TRANS_FRONT_DIFF].feedBack);
58  trans->differential[TRANS_CENTRAL_DIFF].outAxis[0] = &(trans->differential[TRANS_FRONT_DIFF].in);
59 
60  trans->differential[TRANS_CENTRAL_DIFF].inAxis[1] = &(trans->differential[TRANS_REAR_DIFF].feedBack);
61  trans->differential[TRANS_CENTRAL_DIFF].outAxis[1] = &(trans->differential[TRANS_REAR_DIFF].in);
62 
63  if (strcmp(VAL_TRANS_RWD, transType) == 0) {
64  SimDifferentialConfig(hdle, SECT_REARDIFFERENTIAL, &(trans->differential[TRANS_REAR_DIFF]));
65  trans->type = TRANS_RWD;
66  fRatio = trans->differential[TRANS_REAR_DIFF].ratio;
67  //fEff = trans->differential[TRANS_REAR_DIFF].efficiency;
68  } else if (strcmp(VAL_TRANS_FWD, transType) == 0) {
69  SimDifferentialConfig(hdle, SECT_FRNTDIFFERENTIAL, &(trans->differential[TRANS_FRONT_DIFF]));
70  trans->type = TRANS_FWD;
71  fRatio = trans->differential[TRANS_FRONT_DIFF].ratio;
72  //fEff = trans->differential[TRANS_FRONT_DIFF].efficiency;
73  } else if (strcmp(VAL_TRANS_4WD, transType) == 0) {
74  SimDifferentialConfig(hdle, SECT_FRNTDIFFERENTIAL, &(trans->differential[TRANS_FRONT_DIFF]));
75  SimDifferentialConfig(hdle, SECT_REARDIFFERENTIAL, &(trans->differential[TRANS_REAR_DIFF]));
76  SimDifferentialConfig(hdle, SECT_CENTRALDIFFERENTIAL, &(trans->differential[TRANS_CENTRAL_DIFF]));
77  trans->type = TRANS_4WD;
78  fRatio = trans->differential[TRANS_CENTRAL_DIFF].ratio;
79  //fEff = trans->differential[TRANS_FRONT_DIFF].efficiency * trans->differential[TRANS_CENTRAL_DIFF].efficiency * trans->differential[TRANS_REAR_DIFF].efficiency;
80  }
81 
82  trans->gearbox.gearMax = 0;
83  for (i = MAX_GEARS - 1; i >= 0; i--) {
84  snprintf(path, BUFSIZE, "%s/%s/%s", SECT_GEARBOX, ARR_GEARS, gearname[i]);
85  gRatio = GfParmGetNum(hdle, path, PRM_RATIO, (char*)NULL, 0.0f);
86  if ((trans->gearbox.gearMax == 0) && (gRatio != 0.0f)) {
87  trans->gearbox.gearMax = i - 1;
88  }
89  if (gRatio == 0.0f) {
90  carElt->priv.gearRatio[i] = trans->overallRatio[i] = 0.0f;
91  trans->freeI[i] = trans->driveI[i] = 0;
92  trans->gearEff[i] = 1.0;
93  continue;
94  }
95  carElt->priv.gearRatio[i] = trans->overallRatio[i] = gRatio * fRatio;
96  gEff = GfParmGetNum(hdle, path, PRM_EFFICIENCY, (char*)NULL, 1.0f);
97  if (gEff > 1.0f) gEff = 1.0f;
98  if (gEff < 0.0f) gEff = 0.0f;
99  gearI = GfParmGetNum(hdle, path, PRM_INERTIA, (char*)NULL, 0.0f);
100  trans->driveI[i] = (car->engine.I + gearI) * (gRatio * gRatio * fRatio * fRatio);
101  trans->freeI[i] = gearI * (gRatio * gRatio * fRatio * fRatio);
102  trans->gearEff[i] = gEff;
103  }
104 
105  if (gRatio == 0) {
106  /* no reverse */
107  trans->gearbox.gearMin = 0;
108  carElt->priv.gearOffset = 0;
109  } else {
110  trans->gearbox.gearMin = -1;
111  carElt->priv.gearOffset = 1;
112  }
113 
114  carElt->priv.gearNb = trans->gearbox.gearMax + 1;
115 
116  /* initial state */
117  clutch->state = CLUTCH_RELEASING;
118  clutch->timeToRelease = 0;
119  trans->gearbox.gear = 0; /* neutral */
120  trans->curI = trans->freeI[1];
121  switch(trans->type) {
122  case TRANS_RWD:
123  differential = &(trans->differential[TRANS_REAR_DIFF]);
124  differential->outAxis[0]->I = trans->curI / 2.0 + differential->inAxis[0]->I / trans->gearEff[trans->gearbox.gear+1];
125  differential->outAxis[1]->I = trans->curI / 2.0 + differential->inAxis[1]->I / trans->gearEff[trans->gearbox.gear+1];
126  differential->outAxis[0]->Tq = 0;
127  differential->outAxis[1]->Tq = 0;
128  break;
129  case TRANS_FWD:
130  differential = &(trans->differential[TRANS_FRONT_DIFF]);
131  differential->outAxis[0]->I = trans->curI / 2.0 + differential->inAxis[0]->I / trans->gearEff[trans->gearbox.gear+1];
132  differential->outAxis[1]->I = trans->curI / 2.0 + differential->inAxis[1]->I / trans->gearEff[trans->gearbox.gear+1];
133  differential->outAxis[0]->Tq = 0;
134  differential->outAxis[1]->Tq = 0;
135  break;
136  case TRANS_4WD:
137  differential = &(trans->differential[TRANS_FRONT_DIFF]);
138  differential->outAxis[0]->I = trans->curI / 4.0 + differential->inAxis[0]->I / trans->gearEff[trans->gearbox.gear+1];
139  differential->outAxis[1]->I = trans->curI / 4.0 + differential->inAxis[1]->I / trans->gearEff[trans->gearbox.gear+1];
140  differential->outAxis[0]->Tq = 0;
141  differential->outAxis[1]->Tq = 0;
142  differential = &(trans->differential[TRANS_REAR_DIFF]);
143  differential->outAxis[0]->I = trans->curI / 4.0 + differential->inAxis[0]->I / trans->gearEff[trans->gearbox.gear+1];
144  differential->outAxis[1]->I = trans->curI / 4.0 + differential->inAxis[1]->I / trans->gearEff[trans->gearbox.gear+1];
145  differential->outAxis[0]->Tq = 0;
146  differential->outAxis[1]->Tq = 0;
147  differential = &(trans->differential[TRANS_CENTRAL_DIFF]);
148  differential->outAxis[0]->I = trans->curI / 2.0 + differential->inAxis[0]->I / trans->gearEff[trans->gearbox.gear+1];

150  differential->outAxis[0]->Tq = 0;
151  differential->outAxis[1]->Tq = 0;
152  break;
153  }

172 void SimTransmissionReConfig(tCar *car)
173 {
174  void *hdle = car->params;
175  tCarElt *carElt = car->carElt;
176  tTransmission *trans = &(car->transmission);
177  int i;
178  tdble fRatio = 0.0f;
179  const int BUFSIZE = 256;
180  char path[BUFSIZE];
181 
182  if (trans->type == TRANS_RWD) {
183  SimDifferentialReConfig(car, TRANS_REAR_DIFF);
184  fRatio = trans->differential[TRANS_REAR_DIFF].ratio;
185  } else if (trans->type == TRANS_FWD) {
186  SimDifferentialReConfig(car, TRANS_FRONT_DIFF);
187  fRatio = trans->differential[TRANS_FRONT_DIFF].ratio;
188  } else if (trans->type == TRANS_4WD) {
189  SimDifferentialReConfig(car, TRANS_FRONT_DIFF);
190  SimDifferentialReConfig(car, TRANS_REAR_DIFF);
191  SimDifferentialReConfig(car, TRANS_CENTRAL_DIFF);
192  fRatio = trans->differential[TRANS_CENTRAL_DIFF].ratio;
193  }
194 
195  for (i = MAX_GEARS - 1; i - 2 >= 0; i--) {
196  if (trans->overallRatio[i] > 0.0f) {
197  tCarPitSetupValue* v = &car->carElt->pitcmd.setup.gearsratio[i-2];
198  SimAdjustPitCarSetupParam(v);
199  tdble gRatio = v->value;
200  setupGear(hdle, car, carElt, trans, gRatio, fRatio, i);
201  }
202  }
203 
204  // Handle reverse gear
205  if (trans->overallRatio[0] != 0.0f) {
206  snprintf(path, BUFSIZE, "%s/%s/%s", SECT_GEARBOX, ARR_GEARS, gearname[0]);
207  tdble gRatio = GfParmGetNum(hdle, path, PRM_RATIO, (char*)NULL, 0.0f);
208  setupGear(hdle, car, carElt, trans, gRatio, fRatio, 0);
209  }
210 
211  trans->gearbox.gear = 0;
212 }

215 void SimGearboxUpdate(tCar *car)
216 {
217  /* manages gear change */
218  tTransmission *trans = &(car->transmission);
219  tClutch *clutch = &(trans->clutch);
220  tGearbox *gearbox = &(trans->gearbox);
221  tDifferential *differential = NULL;
222 
223  switch(trans->type) {
224  case TRANS_RWD:
225  differential = &(trans->differential[TRANS_REAR_DIFF]);
226  break;
227  case TRANS_FWD:
228  differential = &(trans->differential[TRANS_FRONT_DIFF]);
229  break;
230  case TRANS_4WD:
231  differential = &(trans->differential[TRANS_CENTRAL_DIFF]);
232  break;
233  }
234 
235  trans->curI = trans->driveI[gearbox->gear + 1] * clutch->transferValue + trans->freeI[gearbox->gear + 1] * (1.0f - clutch->transferValue);
236  if (clutch->state == CLUTCH_RELEASING) {
237  clutch->timeToRelease -= SimDeltaTime;
238  if (clutch->timeToRelease <= 0.0f) {
239  clutch->state = CLUTCH_RELEASED;
240  } else {
241  // If user does not engage clutch, we do it automatically.
242  if (clutch->transferValue > 0.99f) {
243  clutch->transferValue = 0.0f;
244  trans->curI = trans->freeI[gearbox->gear + 1];
245  if (car->ctrl->accelCmd > 0.1f) {
246  car->ctrl->accelCmd = 0.1f;
247  }
248  }
249  }
250  } else if ((car->ctrl->gear > gearbox->gear)) {
251  if (car->ctrl->gear <= gearbox->gearMax) {
252  gearbox->gear = car->ctrl->gear;
253  clutch->state = CLUTCH_RELEASING;
254  if (gearbox->gear != 0) {
255  clutch->timeToRelease = clutch->releaseTime;
256  } else {
257  clutch->timeToRelease = 0;
258  }
259  trans->curOverallRatio = trans->overallRatio[gearbox->gear+1];
260  trans->curI = trans->freeI[gearbox->gear+1];
261  differential->in.I = trans->curI + differential->feedBack.I / trans->gearEff[gearbox->gear+1];
262  differential->outAxis[0]->I = trans->curI / 2.0 + differential->inAxis[0]->I / trans->gearEff[gearbox->gear+1];
263  differential->outAxis[1]->I = trans->curI / 2.0 + differential->inAxis[1]->I / trans->gearEff[gearbox->gear+1];
264  if (trans->type == TRANS_4WD) {
265  differential = &(trans->differential[TRANS_FRONT_DIFF]);
266  differential->outAxis[0]->I = trans->curI / 4.0 + differential->inAxis[0]->I / trans->gearEff[gearbox->gear+1];
267  differential->outAxis[1]->I = trans->curI / 4.0 + differential->inAxis[1]->I / trans->gearEff[gearbox->gear+1];
268  differential = &(trans->differential[TRANS_REAR_DIFF]);
269  differential->outAxis[0]->I = trans->curI / 4.0 + differential->inAxis[0]->I / trans->gearEff[gearbox->gear+1];
270  differential->outAxis[1]->I = trans->curI / 4.0 + differential->inAxis[1]->I / trans->gearEff[gearbox->gear+1];
271  }
272  }
273  } else if ((car->ctrl->gear < gearbox->gear)) {
274  if (car->ctrl->gear >= gearbox->gearMin) {
275  gearbox->gear = car->ctrl->gear;
276  clutch->state = CLUTCH_RELEASING;
277  if (gearbox->gear != 0) {
278  clutch->timeToRelease = clutch->releaseTime;
279  } else {
280  clutch->timeToRelease = 0;
281  }
282  trans->curOverallRatio = trans->overallRatio[gearbox->gear+1];
283  trans->curI = trans->freeI[gearbox->gear+1];
284  differential->in.I = trans->curI + differential->feedBack.I / trans->gearEff[gearbox->gear+1];
285  differential->outAxis[0]->I = trans->curI / 2.0 + differential->inAxis[0]->I / trans->gearEff[gearbox->gear+1];
286  differential->outAxis[1]->I = trans->curI / 2.0 + differential->inAxis[1]->I / trans->gearEff[gearbox->gear+1];
287  if (trans->type == TRANS_4WD) {
288  differential = &(trans->differential[TRANS_FRONT_DIFF]);
289  differential->outAxis[0]->I = trans->curI / 4.0 + differential->inAxis[0]->I / trans->gearEff[gearbox->gear+1];
290  differential->outAxis[1]->I = trans->curI / 4.0 + differential->inAxis[1]->I / trans->gearEff[gearbox->gear+1];
291  differential = &(trans->differential[TRANS_REAR_DIFF]);
292  differential->outAxis[0]->I = trans->curI / 4.0 + differential->inAxis[0]->I / trans->gearEff[gearbox->gear+1];
293  differential->outAxis[1]->I = trans->curI / 4.0 + differential->inAxis[1]->I / trans->gearEff[gearbox->gear+1];
294  }
295  }
296  }

300 SimTransmissionUpdate(tCar *car)
301 {
302  tTransmission *trans = &(car->transmission);
303  tClutch *clutch = &(trans->clutch);
304  tDifferential *differential, *differential0, *differential1;
305  tdble transfer = MIN(clutch->transferValue * 3.0, 1.0);
306 
307  switch(trans->type) {
308  case TRANS_RWD:
309  differential = &(trans->differential[TRANS_REAR_DIFF]);
310  differential->in.Tq = car->engine.Tq * trans->curOverallRatio * transfer;
311  SimDifferentialUpdate(car, differential, 1);
312  SimUpdateFreeWheels(car, 0);
313  break;
314  case TRANS_FWD:
315  differential = &(trans->differential[TRANS_FRONT_DIFF]);
316  differential->in.Tq = car->engine.Tq * trans->curOverallRatio * transfer;
317  SimDifferentialUpdate(car, differential, 1);
318  SimUpdateFreeWheels(car, 1);
319  break;
320  case TRANS_4WD:
321  differential = &(trans->differential[TRANS_CENTRAL_DIFF]);
322  differential0 = &(trans->differential[TRANS_FRONT_DIFF]);
323  differential1 = &(trans->differential[TRANS_REAR_DIFF]);
324 
325  differential->in.Tq = car->engine.Tq * trans->curOverallRatio * transfer;
326  differential->inAxis[0]->spinVel = (differential0->inAxis[0]->spinVel + differential0->inAxis[1]->spinVel) / 2.0;
327  differential->inAxis[1]->spinVel = (differential1->inAxis[0]->spinVel + differential1->inAxis[1]->spinVel) / 2.0;
328  differential->inAxis[0]->Tq = (differential0->inAxis[0]->Tq + differential0->inAxis[1]->Tq) / differential->ratio;
329  differential->inAxis[1]->Tq = (differential1->inAxis[0]->Tq + differential1->inAxis[1]->Tq) / differential->ratio;
330  differential->inAxis[0]->brkTq = (differential0->inAxis[0]->brkTq + differential0->inAxis[1]->brkTq) / differential->ratio;
331  differential->inAxis[1]->brkTq = (differential1->inAxis[0]->brkTq + differential1->inAxis[1]->brkTq) / differential->ratio;
332 
333  SimDifferentialUpdate(car, differential, 1);
334 
335  differential = differential0;
336  SimDifferentialUpdate(car, differential, 0);
337 
338  differential = differential1;
339  SimDifferentialUpdate(car, differential, 0);
340  break;
341  }
