22 void SimDifferentialConfig(void *hdle, const char *section, tDifferential *differential)
23 {
24  differential->I = GfParmGetNum(hdle, section, PRM_INERTIA, (char*)NULL, 0.1f);
25  differential->efficiency = GfParmGetNum(hdle, section, PRM_EFFICIENCY, (char*)NULL, 1.0f);
26  differential->ratio = GfParmGetNum(hdle, section, PRM_RATIO, (char*)NULL, 1.0f);
27  differential->dTqMin = GfParmGetNum(hdle, section, PRM_MIN_TQ_BIAS, (char*)NULL, 0.05f);
28  differential->dTqMax = GfParmGetNum(hdle, section, PRM_MAX_TQ_BIAS, (char*)NULL, 0.80f) - differential->dTqMin;
29  if (differential->dTqMax < 0.0f) differential->dTqMax = 0.0f;
30  differential->dSlipMax = GfParmGetNum(hdle, section, PRM_MAX_SLIP_BIAS, (char*)NULL, 0.03f);
31  differential->lockInputTq = GfParmGetNum(hdle, section, PRM_LOCKING_TQ, (char*)NULL, 3000.0f);
32  differential->lockBrakeInputTq = GfParmGetNum(hdle, section, PRM_LOCKINGBRAKE_TQ, (char*)NULL, differential->lockInputTq*0.33f);
33  differential->viscosity = GfParmGetNum(hdle, section, PRM_VISCOSITY_FACTOR, (char*)NULL, 1.0f);
34 
35  const char* type = GfParmGetStr(hdle, section, PRM_TYPE, VAL_DIFF_NONE);
36  if (strcmp(type, VAL_DIFF_LIMITED_SLIP) == 0) {
37  differential->type = DIFF_LIMITED_SLIP;
38  } else if (strcmp(type, VAL_DIFF_VISCOUS_COUPLER) == 0) {
39  differential->type = DIFF_VISCOUS_COUPLER;
40  } else if (strcmp(type, VAL_DIFF_SPOOL) == 0) {
41  differential->type = DIFF_SPOOL;
42  } else if (strcmp(type, VAL_DIFF_FREE) == 0) {
43  differential->type = DIFF_FREE;
44  } else {
45  differential->type = DIFF_NONE;
46  }
47 
48  differential->feedBack.I = differential->I * differential->ratio * differential->ratio +
49  (differential->inAxis[0]->I + differential->inAxis[1]->I) / differential->efficiency;
50 }

53 void SimDifferentialReConfig(tCar* car, int index)
54 {
55  tDifferential *differential = &car->transmission.differential[index];
56 
57  // Ratio
58  tCarPitSetupValue* v = &car->carElt->pitcmd.setup.diffratio[index];
59  if (SimAdjustPitCarSetupParam(v)) {
60  differential->ratio = v->value;
61  differential->feedBack.I = differential->I * differential->ratio * differential->ratio +
62  (differential->inAxis[0]->I + differential->inAxis[1]->I) / differential->efficiency;
63  }
64 
65  // Min torque bias
66  v = &car->carElt->pitcmd.setup.diffmintqbias[index];
67  if (SimAdjustPitCarSetupParam(v)) {
68  differential->dTqMin = v->value;
69  }
70 
71  // Max torque bias
72  v = &car->carElt->pitcmd.setup.diffmaxtqbias[index];
73  if (SimAdjustPitCarSetupParam(v)) {
74  differential->dTqMax = v->value - differential->dTqMin;
75  if (differential->dTqMax < 0.0f) {
76  differential->dTqMax = 0.0f;
77  v->value = differential->dTqMin;
78  }
79  }
80 
81  // Slip bias
82  v = &car->carElt->pitcmd.setup.diffslipbias[index];
83  if (SimAdjustPitCarSetupParam(v)) {
84  differential->dSlipMax = v->value;
85  }
86 
87  // Locking input torque
88  v = &car->carElt->pitcmd.setup.difflockinginputtq[index];
89  if (SimAdjustPitCarSetupParam(v)) {
90  differential->lockInputTq = v->value;
91  }
92 
93  // Locking brake input torque
94  v = &car->carElt->pitcmd.setup.difflockinginputbraketq[index];
95  if (SimAdjustPitCarSetupParam(v)) {
96  differential->lockBrakeInputTq = v->value;
97  }
98 }

101 static void updateSpool(tCar *car, tDifferential *differential, int first)
102 {
103  tdble DrTq;
104  tdble ndot;
105  tdble spinVel;
106  tdble BrTq;
107  tdble engineReaction;
108  tdble I;
109  tdble inTq, brkTq;
110 
111  DrTq = differential->in.Tq;
112 
113  I = differential->outAxis[0]->I + differential->outAxis[1]->I;
114  inTq = differential->inAxis[0]->Tq + differential->inAxis[1]->Tq;
115  brkTq = differential->inAxis[0]->brkTq + differential->inAxis[1]->brkTq;
116 
117  ndot = SimDeltaTime * (DrTq - inTq) / I;
118  spinVel = differential->inAxis[0]->spinVel + ndot;
119 
120  BrTq = - SIGN(spinVel) * brkTq;
121  ndot = SimDeltaTime * BrTq / I;
122 
123  if (((ndot * spinVel) < 0.0) && (fabs(ndot) > fabs(spinVel))) {
124  ndot = -spinVel;
125  }
126  if ((spinVel == 0.0) && (ndot < 0.0)) ndot = 0;
127 
128  spinVel += ndot;
129  if (first) {
130  engineReaction = SimEngineUpdateRpm(car, spinVel);
131  if (engineReaction != 0.0) {
132  spinVel = engineReaction;
133  }
134  }
135  differential->outAxis[0]->spinVel = differential->outAxis[1]->spinVel = spinVel;
136 
137  differential->outAxis[0]->Tq = (differential->outAxis[0]->spinVel - differential->inAxis[0]->spinVel) / SimDeltaTime * differential->outAxis[0]->I;
138  differential->outAxis[1]->Tq = (differential->outAxis[1]->spinVel - differential->inAxis[1]->spinVel) / SimDeltaTime * differential->outAxis[1]->I;
139 }

142 void
143 SimDifferentialUpdate(tCar *car, tDifferential *differential, int first)
144 {
145  tdble DrTq, DrTq0, DrTq1;
146  tdble ndot0, ndot1;
147  tdble spinVel0, spinVel1;
148  tdble inTq0, inTq1;
149  tdble spdRatioMax, commomSpinVel;
150  tdble deltaSpd, deltaTq, bias, lockTq, biassign;
151  tdble BrTq;
152  tdble engineReaction;
153  tdble meanv;
154 
155  if (differential->type == DIFF_SPOOL) {
156  updateSpool(car, differential, first);
157  return;
158  }
159 
160  DrTq = differential->in.Tq;
161 
162  spinVel0 = differential->inAxis[0]->spinVel;
163  spinVel1 = differential->inAxis[1]->spinVel;
164 
165  inTq0 = differential->inAxis[0]->Tq;
166  inTq1 = differential->inAxis[1]->Tq;
167 
168  commomSpinVel = fabs(spinVel0) + fabs(spinVel1);
169  if (commomSpinVel != 0) {
170  tdble spdRatio = fabs(spinVel0 - spinVel1) / commomSpinVel;
171 
172  switch (differential->type) {
173  case DIFF_FREE:
174  {
175  float spiderTq = inTq1 - inTq0;
176  DrTq0 = (DrTq + spiderTq)*0.5f;
177  DrTq1 = (DrTq - spiderTq)*0.5f;
178  }
179  break;
180  case DIFF_LIMITED_SLIP:
181  if (DrTq > differential->lockInputTq || DrTq < -differential->lockBrakeInputTq) {
182  updateSpool(car, differential, first);
183  return;
184  }
185 
186  if (DrTq >= 0.0f) {
187  lockTq = differential->lockInputTq;
188  biassign = 1.0f;
189  } else {
190  lockTq = -differential->lockBrakeInputTq;
191  biassign = -1.0f;
192  }
193 
194  spdRatioMax = differential->dSlipMax - DrTq * differential->dSlipMax / lockTq;
195  bias = 0.0f;
196  if (spdRatio > spdRatioMax) {
197  deltaSpd = (spdRatio - spdRatioMax) * commomSpinVel / 2.0;
198  if (spinVel0 > spinVel1) {
199  spinVel0 -= deltaSpd;
200  spinVel1 += deltaSpd;
201  bias = -(spdRatio - spdRatioMax);
202  } else {
203  spinVel0 += deltaSpd;
204  spinVel1 -= deltaSpd;
205  bias = (spdRatio - spdRatioMax);
206  }
207  }
208 
209  {
210  float spiderTq = inTq1 - inTq0;
211  DrTq0 = (DrTq*(1.0f + bias*biassign) + spiderTq)*0.5f;
212  DrTq1 = (DrTq*(1.0f - bias*biassign) - spiderTq)*0.5f;
213  }
214  break;
215  case DIFF_VISCOUS_COUPLER:
216  if (spinVel0 >= spinVel1) {
217  DrTq0 = DrTq * differential->dTqMin;
218  DrTq1 = DrTq * (1 - differential->dTqMin);
219  } else {
220  deltaTq = differential->dTqMin + (1.0 - exp(-fabs(differential->viscosity * (spinVel0 - spinVel1)))) * differential->dTqMax;
221  DrTq0 = DrTq * deltaTq;
222  DrTq1 = DrTq * (1 - deltaTq);
223  }
224 
225  break;
226  default: /* NONE ? */
227  DrTq0 = DrTq1 = 0;
228  break;
229  }
230  } else {
231  DrTq0 = DrTq / 2.0;
232  DrTq1 = DrTq / 2.0;
233  }
234 
235  ndot0 = SimDeltaTime * (DrTq0 - inTq0) / differential->outAxis[0]->I;
236  spinVel0 += ndot0;
237  ndot1 = SimDeltaTime * (DrTq1 - inTq1) / differential->outAxis[1]->I;
238  spinVel1 += ndot1;
239 
240  BrTq = - SIGN(spinVel0) * differential->inAxis[0]->brkTq;
241  ndot0 = SimDeltaTime * BrTq / differential->outAxis[0]->I;
242  if (((ndot0 * spinVel0) < 0.0) && (fabs(ndot0) > fabs(spinVel0))) {
243  ndot0 = -spinVel0;
244  }
245  if ((spinVel0 == 0.0) && (ndot0 < 0.0)) ndot0 = 0;
246  spinVel0 += ndot0;
247 
248  BrTq = - SIGN(spinVel1) * differential->inAxis[1]->brkTq;
249  ndot1 = SimDeltaTime * BrTq / differential->outAxis[1]->I;
250  if (((ndot1 * spinVel1) < 0.0) && (fabs(ndot1) > fabs(spinVel1))) {
251  ndot1 = -spinVel1;
252  }
253  if ((spinVel1 == 0.0) && (ndot1 < 0.0)) ndot1 = 0;
254  spinVel1 += ndot1;
255 
256  if (first) {
257  meanv = (spinVel0 + spinVel1) / 2.0;
258  engineReaction = SimEngineUpdateRpm(car, meanv);
259  if (meanv != 0.0) {
260  engineReaction = engineReaction / meanv;
261  if (engineReaction != 0.0) {
262  spinVel1 *= engineReaction;
263  spinVel0 *= engineReaction;
264  }
265  }
266  }
267 
268  differential->outAxis[0]->spinVel = spinVel0;
269  differential->outAxis[1]->spinVel = spinVel1;
270 
271  differential->outAxis[0]->Tq = (differential->outAxis[0]->spinVel - differential->inAxis[0]->spinVel) / SimDeltaTime * differential->outAxis[0]->I;
272  differential->outAxis[1]->Tq = (differential->outAxis[1]->spinVel - differential->inAxis[1]->spinVel) / SimDeltaTime * differential->outAxis[1]->I;
273 }
