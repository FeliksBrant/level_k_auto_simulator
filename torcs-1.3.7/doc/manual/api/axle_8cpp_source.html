24 void SimAxleConfig(tCar *car, int index)
25 {
26  void *hdle = car->params;
27  tdble rollCenter;
28 
29  tAxle *axle = &(car->axle[index]);
30 
31  axle->xpos = GfParmGetNum(hdle, AxleSect[index], PRM_XPOS, (char*)NULL, 0.0f);
32  axle->I = GfParmGetNum(hdle, AxleSect[index], PRM_INERTIA, (char*)NULL, 0.15f);
33  rollCenter = GfParmGetNum(hdle, AxleSect[index], PRM_ROLLCENTER, (char*)NULL, 0.15f);
34  car->wheel[index*2].rollCenter = car->wheel[index*2+1].rollCenter = rollCenter;
35 
36  tdble x0 = GfParmGetNum(hdle, AxleSect[index], PRM_SUSPCOURSE, (char*)NULL, 0.0f);
37  SimSuspConfig(hdle, AxleSect[index], &(axle->thirdSusp), 0.0f, x0);
38 
39  if (index == 0) {
40  axle->arbSuspSpringK = GfParmGetNum(hdle, SECT_FRNTARB, PRM_SPR, (char*)NULL, 0.0f);
41  } else {
42  axle->arbSuspSpringK = GfParmGetNum(hdle, SECT_REARARB, PRM_SPR, (char*)NULL, 0.0f);
43  }
44 
45  car->wheel[index*2].feedBack.I += axle->I / 2.0;
46  car->wheel[index*2+1].feedBack.I += axle->I / 2.0;
47 }

67 void SimAxleUpdate(tCar *car, int index)
68 {
69  tAxle *axle = &(car->axle[index]);
70  tdble str, stl, vr, vl;
71 
72  str = car->wheel[index*2].susp.x;
73  stl = car->wheel[index*2+1].susp.x;
74  vr = car->wheel[index*2].susp.v;
75  vl = car->wheel[index*2+1].susp.v;
76 
77  // Anti roll bar
78  tdble farb = axle->arbSuspSpringK * (stl - str);
79 
80  // Third element
81  axle->thirdSusp.x = (stl + str)/2.0f;
82  axle->thirdSusp.v = (vl + vr)/2.0f;
83  SimSuspUpdate(&axle->thirdSusp);
84  tdble fthird = 0.0f;
85  if (axle->thirdSusp.x < axle->thirdSusp.spring.xMax && axle->thirdSusp.force > 0.0f) {
86  fthird = axle->thirdSusp.force/2.0f;
87  }
88 
89  // right
90  car->wheel[index*2].axleFz = farb + fthird;
91  // left
92  car->wheel[index*2+1].axleFz = -farb + fthird;
93 }
