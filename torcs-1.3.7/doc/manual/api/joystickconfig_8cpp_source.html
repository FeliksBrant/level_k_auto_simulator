115  case 3:
116  case 4:

133 static void Idle2(void)
134 {
135  int mask;
136  int b, i;
137  int index;
138 
139  for (index = 0; index < NUM_JOY; index++) {
140  if (js[index]) {
141  js[index]->read(&b, &ax[index * _JS_MAX_AXES]);
142 
143  /* Joystick buttons */
144  for (i = 0, mask = 1; i < 32; i++, mask *= 2) {
145  if (((b & mask) != 0) && ((rawb[index] & mask) == 0)) {
146  const char *str = GfctrlGetNameByRef(GFCTRL_TYPE_JOY_BUT, i + 32 * index);
147  if (!GfctrlIsEventBlacklisted(parmHandle, driverSection, str)) {
148  /* Button fired */
149  JoyCalAutomaton();
150  if (CalState >= NB_STEPS) {
151  glutIdleFunc(GfuiIdle);
152  }
153  glutPostRedisplay();
154  rawb[index] = b;
155  return;
156  }
157  }
158  }
159  rawb[index] = b;
160  }
161  }
162 }
