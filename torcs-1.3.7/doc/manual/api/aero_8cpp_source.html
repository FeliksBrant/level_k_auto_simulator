38 void SimAeroUpdate(tCar *car, tSituation *s)
39 {
40  tdble hm;
41  int i;
42  tCar *otherCar;
43  tdble x, y;
44  tdble yaw, otherYaw, airSpeed, tmpas, spdang, tmpsdpang, dyaw;
45  tdble dragK = 1.0;
46 
47  x = car->DynGCg.pos.x;
48  y = car->DynGCg.pos.y;
49  yaw = car->DynGCg.pos.az;
50  airSpeed = car->DynGC.vel.x;
51  spdang = atan2(car->DynGCg.vel.y, car->DynGCg.vel.x);
52 
53  if (airSpeed > 10.0f) {
54  for (i = 0; i < s->_ncars; i++) {
55  if (i == car->carElt->index) {
56  // skip myself
57  continue;
58  }
59 
60  otherCar = &(SimCarTable[i]);
61  otherYaw = otherCar->DynGCg.pos.az;
62  tmpsdpang = spdang - atan2(y - otherCar->DynGCg.pos.y, x - otherCar->DynGCg.pos.x);
63  NORM_PI_PI(tmpsdpang);
64  dyaw = yaw - otherYaw;
65  NORM_PI_PI(dyaw);
66 
67  if ((otherCar->DynGC.vel.x > 10.0f) && (fabs(dyaw) < 0.1396f)) {
68  if (fabs(tmpsdpang) > 2.9671f) { /* 10 degrees */
69  // behind another car
70  tmpas = 1.0f - exp(- 2.0f * DIST(x, y, otherCar->DynGCg.pos.x, otherCar->DynGCg.pos.y) /
71  (otherCar->aero.Cd * otherCar->DynGC.vel.x));
72  if (tmpas < dragK) {
73  dragK = tmpas;
74  }
75  } else if (fabs(tmpsdpang) < 0.1396f) { /* 8 degrees */
76  // before another car, lower drag by maximum 15% (this is just another guess)
77  tmpas = 1.0f - 0.15f * exp(- 8.0f * DIST(x, y, otherCar->DynGCg.pos.x, otherCar->DynGCg.pos.y) / (car->aero.Cd * car->DynGC.vel.x));
78  if (tmpas < dragK) {
79  dragK = tmpas;
80  }
81  }
82  }
83  }
84  }
85 
86  car->airSpeed2 = airSpeed * airSpeed;
87  tdble v2 = car->airSpeed2;
88 
89  // simulate ground effect drop off caused by non-frontal airflow (diffusor stops working etc.)
90  tdble cosa = 1.0f;
91  if (car->speed > 1.0f) {
92  cosa = car->DynGC.vel.x/car->speed;
93  }
94 
95  if (cosa < 0.0f) {
96  cosa = 0.0f;
97  }
98 
99  car->aero.drag = -SIGN(car->DynGC.vel.x) * car->aero.SCx2 * v2 * (1.0f + (tdble)car->dammage / 10000.0f) * dragK * dragK;
100 
101  hm = 1.5f * (car->wheel[0].rideHeight + car->wheel[1].rideHeight + car->wheel[2].rideHeight + car->wheel[3].rideHeight);
102  hm = hm*hm;
103  hm = hm*hm;
104  hm = 2.0f * exp(-3.0f*hm);
105  car->aero.lift[0] = - car->aero.Clift[0] * v2 * hm * cosa;
106  car->aero.lift[1] = - car->aero.Clift[1] * v2 * hm * cosa;
107 }
