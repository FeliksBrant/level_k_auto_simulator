27 void SimWheelConfig(tCar *car, int index)
28 {
29  void *hdle = car->params;
30  tCarElt *carElt = car->carElt;
31  tWheel *wheel = &(car->wheel[index]);
32  tdble rimdiam, tirewidth, tireratio, pressure;
33  tdble x0, Ca, RFactor, EFactor, patchLen;
34 
35  pressure = GfParmGetNum(hdle, WheelSect[index], PRM_PRESSURE, (char*)NULL, 275600);
36  rimdiam = GfParmGetNum(hdle, WheelSect[index], PRM_RIMDIAM, (char*)NULL, 0.33f);
37  tirewidth = GfParmGetNum(hdle, WheelSect[index], PRM_TIREWIDTH, (char*)NULL, 0.145f);
38  tireratio = GfParmGetNum(hdle, WheelSect[index], PRM_TIRERATIO, (char*)NULL, 0.75f);
39  wheel->mu = GfParmGetNum(hdle, WheelSect[index], PRM_MU, (char*)NULL, 1.0f);
40  wheel->I = GfParmGetNum(hdle, WheelSect[index], PRM_INERTIA, (char*)NULL, 1.5f);
41  wheel->I += wheel->brake.I; // add brake inertia
42  wheel->staticPos.y = GfParmGetNum(hdle, WheelSect[index], PRM_YPOS, (char*)NULL, 0.0f);
43  x0 = GfParmGetNum(hdle, WheelSect[index], PRM_RIDEHEIGHT, (char*)NULL, 0.20f);
44  wheel->staticPos.az = GfParmGetNum(hdle, WheelSect[index], PRM_TOE, (char*)NULL, 0.0f);
45  wheel->staticPos.ax = GfParmGetNum(hdle, WheelSect[index], PRM_CAMBER, (char*)NULL, 0.0f);
46  Ca = GfParmGetNum(hdle, WheelSect[index], PRM_CA, (char*)NULL, 30.0f);
47  RFactor = GfParmGetNum(hdle, WheelSect[index], PRM_RFACTOR, (char*)NULL, 0.8f);
48  EFactor = GfParmGetNum(hdle, WheelSect[index], PRM_EFACTOR, (char*)NULL, 0.7f);
49  wheel->lfMax = GfParmGetNum(hdle, WheelSect[index], PRM_LOADFMAX, (char*)NULL, 1.6f);
50  wheel->lfMin = GfParmGetNum(hdle, WheelSect[index], PRM_LOADFMIN, (char*)NULL, 0.8f);
51  wheel->opLoad = GfParmGetNum(hdle, WheelSect[index], PRM_OPLOAD, (char*)NULL, wheel->weight0 * 1.2f);
52  wheel->mass = GfParmGetNum(hdle, WheelSect[index], PRM_MASS, (char*)NULL, 20.0f);
53 
54  if (index % 2) {
55  wheel->relPos.ax = -wheel->staticPos.ax;
56  } else {
57  wheel->relPos.ax = wheel->staticPos.ax;
58  }
59 
60  wheel->lfMin = MIN(0.8f, wheel->lfMin);
61  wheel->lfMax = MAX(1.6f, wheel->lfMax);
62 
63  RFactor = MIN(1.0f, RFactor);
64  RFactor = MAX(0.1f, RFactor);
65  EFactor = MIN(1.0f, EFactor);
66 
67  patchLen = wheel->weight0 / (tirewidth * pressure);
68 
69  wheel->radius = rimdiam / 2.0f + tirewidth * tireratio;
70  wheel->tireSpringRate = wheel->weight0 / (wheel->radius * (1.0f - cos(asin(patchLen / (2.0f * wheel->radius)))));
71  wheel->relPos.x = wheel->staticPos.x = car->axle[index/2].xpos;
72  wheel->relPos.y = wheel->staticPos.y;
73  wheel->relPos.z = wheel->radius - wheel->susp.spring.x0;
74  wheel->relPos.ay = wheel->relPos.az = 0.0f;
75  wheel->steer = 0.0f;
76 
77  /* components */
78  SimSuspConfig(hdle, SuspSect[index], &(wheel->susp), wheel->weight0, x0);
79  SimBrakeConfig(hdle, BrkSect[index], &(wheel->brake));
80 
81  carElt->_rimRadius(index) = rimdiam / 2.0f;
82  carElt->_tireHeight(index) = tirewidth * tireratio;
83  carElt->_tireWidth(index) = tirewidth;
84  carElt->_brakeDiskRadius(index) = wheel->brake.radius;
85  carElt->_wheelRadius(index) = wheel->radius;
86 
87  wheel->mfC = 2.0f - asin(RFactor) * 2.0f / PI;
88  wheel->mfB = Ca / wheel->mfC;
89  wheel->mfE = EFactor;
90 
91  wheel->lfK = log((1.0f - wheel->lfMin) / (wheel->lfMax - wheel->lfMin));
92 
93  wheel->feedBack.I += wheel->I;
94  wheel->feedBack.spinVel = 0.0f;
95  wheel->feedBack.Tq = 0.0f;
96  wheel->feedBack.brkTq = 0.0f;
97  wheel->rel_vel = 0.0f;
98 }

129 void SimWheelUpdateRide(tCar *car, int index)
130 {
131  tWheel *wheel = &(car->wheel[index]);
132  tdble Zroad;
133 
134  // compute suspension travel
135  RtTrackGlobal2Local(car->trkPos.seg, wheel->pos.x, wheel->pos.y, &(wheel->trkPos), TR_LPOS_SEGMENT);
136  wheel->zRoad = Zroad = RtTrackHeightL(&(wheel->trkPos));
137 
138  // Wheel susp.x is not the wheel movement, look at SimSuspCheckIn, it becomes there scaled with
139  // susp->spring.bellcrank, so we invert this here.
140  tdble prexwheel = wheel->susp.x / wheel->susp.spring.bellcrank;
141 
142  tdble new_susp_x= prexwheel - wheel->rel_vel * SimDeltaTime;
143  tdble max_extend = wheel->pos.z - Zroad;
144  wheel->rideHeight = max_extend;
145 
146  wheel->state &= ~SIM_WH_ONAIR;
147  if (max_extend < new_susp_x) {
148  new_susp_x = max_extend;
149  wheel->rel_vel = 0.0f;
150  } else if (new_susp_x < wheel->susp.spring.packers) {
151  new_susp_x = wheel->susp.spring.packers;
152  wheel->rel_vel = 0.0f;
153  }
154 
155  if (new_susp_x < max_extend) {
156  wheel->state |= SIM_WH_ONAIR;
157  }
158 
159  tdble prex = wheel->susp.x;
160  wheel->susp.x = new_susp_x;
161 
162  // verify the suspension travel, beware, wheel->susp.x will be scaled by SimSuspCheckIn
163  SimSuspCheckIn(&(wheel->susp));
164  wheel->susp.v = (prex - wheel->susp.x) / SimDeltaTime;
165 
166  // update wheel brake
167  SimBrakeUpdate(car, wheel, &(wheel->brake));
168 }

173 void SimWheelUpdateForce(tCar *car, int index)
174 {
175  tWheel *wheel = &(car->wheel[index]);
176  tdble axleFz = wheel->axleFz;
177  tdble vt, v, v2, wrl; // wheel related velocity
178  tdble Fn, Ft;
179  tdble waz;
180  tdble CosA, SinA;
181  tdble s, sa, sx, sy; // slip vector
182  tdble stmp, F, Bx;
183  tdble mu;
184  wheel->state = 0;
185 
186  // VERTICAL STUFF CONSIDERING SMALL PITCH AND ROLL ANGLES
187  // update suspension force
188  SimSuspUpdate(&(wheel->susp));
189 
190  // check suspension state
191  wheel->state |= wheel->susp.state;
192  if ((wheel->state & SIM_SUSP_EXT) && wheel->rel_vel <= 0.0f) {
193  wheel->forces.z = wheel->rel_vel/SimDeltaTime*wheel->mass;
194  wheel->rel_vel = 0.0;
195  } else {
196  wheel->forces.z = axleFz + wheel->susp.force;
197  wheel->rel_vel -= SimDeltaTime * wheel->forces.z / wheel->mass;
198  }
199 
200  // update wheel coord, center relative to GC
201  wheel->relPos.z = - wheel->susp.x / wheel->susp.spring.bellcrank + wheel->radius;
202 
203  // Wheel contact with the ground can just push the car upwards, but not drag it towards the ground.
204  // Because forces.z includes the inertia of the wheel, the wheel force can act on the car body
205  // without interacting with the ground.
206  tdble zforce = wheel->forces.z;
207  if ((zforce < 0.0f) || (wheel->state & SIM_WH_ONAIR)) {
208  zforce = 0.0f;
209  }
210 
211  // HORIZONTAL FORCES
212  waz = wheel->steer + wheel->staticPos.az;
213  CosA = cos(waz);
214  SinA = sin(waz);
215 
216  // tangent velocity.
217  vt = wheel->bodyVel.x * CosA + wheel->bodyVel.y * SinA;
218  v2 = wheel->bodyVel.x * wheel->bodyVel.x + wheel->bodyVel.y * wheel->bodyVel.y;
219  v = sqrt(v2);
220 
221  // slip angle
222  if (v < 0.000001f) {
223  sa = 0.0f;
224  } else {
225  sa = atan2(wheel->bodyVel.y, wheel->bodyVel.x) - waz;
226  }
227  NORM_PI_PI(sa);
228 
229  wrl = wheel->spinVel * wheel->radius;
230  if ((wheel->state & SIM_WH_ONAIR) != 0) {
231  sx = sy = 0.0f;
232  } else if (v < 0.000001f) {
233  sx = wrl;
234  sy = 0.0f;
235  } else {
236  sx = (vt - wrl) / fabs(vt);
237  sy = sin(sa);
238  }
239 
240  Ft = 0.0f;
241  Fn = 0.0f;
242  s = sqrt(sx*sx+sy*sy);
243 
244  {
245  // calculate _skid and _reaction for sound.
246  if (v < 2.0f) {
247  car->carElt->_skid[index] = 0.0f;
248  } else {
249  car->carElt->_skid[index] = MIN(1.0f, (s*zforce*0.0002f));
250  }
251  }
252 
253  stmp = MIN(s, 1.5f);
254 
255  // MAGIC FORMULA
256  Bx = wheel->mfB * stmp;
257  F = sin(wheel->mfC * atan(Bx * (1.0f - wheel->mfE) + wheel->mfE * atan(Bx))) * (1.0f + stmp * simSkidFactor[car->carElt->_skillLevel]);
258 
259  // load sensitivity
260  mu = wheel->mu * (wheel->lfMin + (wheel->lfMax - wheel->lfMin) * exp(wheel->lfK * zforce / wheel->opLoad));
261 
262  F *= zforce * mu * wheel->trkPos.seg->surface->kFriction * (1.0f + 0.05f * sin(-wheel->staticPos.ax * 18.0f)); /* coeff */
263 
264  wheel->rollRes = zforce * wheel->trkPos.seg->surface->kRollRes;
265  car->carElt->priv.wheel[index].rollRes = wheel->rollRes;
266 
267  if (s > 0.000001f) {
268  // wheel axis based
269  Ft -= F * sx / s;
270  Fn -= F * sy / s;
271  }
272 
273  RELAXATION2(Fn, wheel->preFn, 50.0f);
274  RELAXATION2(Ft, wheel->preFt, 50.0f);
275 
276  wheel->relPos.az = waz;
277 
278  wheel->forces.x = Ft * CosA - Fn * SinA;
279  wheel->forces.y = Ft * SinA + Fn * CosA;
280  wheel->spinTq = Ft * wheel->radius;
281  wheel->sa = sa;
282  wheel->sx = sx;
283 
284  wheel->feedBack.spinVel = wheel->spinVel;
285  wheel->feedBack.Tq = wheel->spinTq;
286  wheel->feedBack.brkTq = wheel->brake.Tq;
287 
288  car->carElt->_wheelSlipSide(index) = sy*v;
289  car->carElt->_wheelSlipAccel(index) = sx*v;
290  car->carElt->_reaction[index] = zforce;
291 }

314 SimUpdateFreeWheels(tCar *car, int axlenb)
315 {
316  int i;
317  tWheel *wheel;
318  tdble BrTq; // brake torque
319  tdble ndot; // rotation acceleration
320  tdble I;
321 
322  for (i = axlenb * 2; i < axlenb * 2 + 2; i++) {
323  wheel = &(car->wheel[i]);
324 
325  I = wheel->I + car->axle[axlenb].I / 2.0f;
326 
327  ndot = SimDeltaTime * wheel->spinTq / I;
328  wheel->spinVel -= ndot;
329 
330  BrTq = - SIGN(wheel->spinVel) * wheel->brake.Tq;
331  ndot = SimDeltaTime * BrTq / I;
332 
333  if (fabs(ndot) > fabs(wheel->spinVel)) {
334  ndot = -wheel->spinVel;
335  }
336 
337  wheel->spinVel += ndot;
338  wheel->in.spinVel = wheel->spinVel;
339  }
