98  GfctrlMouseGetCurrent(&mouseInfo);

101  return; /* ignore no move input */
102  }
103  Cmd[CMD_OFFSET + CalState].max = axv;
104  Cmd[CMD_OFFSET + CalState].pow = 1.0 / axv;
