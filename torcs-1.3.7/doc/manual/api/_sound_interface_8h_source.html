328 class SharedSourcePool {
329  public:
330  SharedSourcePool(int nbsources):nbsources(nbsources) {
331  pool = new sharedSource[nbsources];
332  int i;
333  for (i = 0; i < nbsources; i++) {
334  pool[i].currentOwner = NULL;
335  pool[i].in_use = false;
336  alGenSources(1, &(pool[i].source));
337  int error = alGetError();
338  if (error != AL_NO_ERROR) {
339  printf("OpenAL error, allocating dynamic source index %d\n", i);
340  this->nbsources = i;
341  break;
342  }
343  }
344  printf(" Dynamic Sources: requested: %d, created: %d\n", nbsources, this->nbsources);
345  }
346 
347  virtual ~SharedSourcePool() {
348  int i;
349  for (i = 0; i < nbsources; i++) {
350  alDeleteSources(1, &(pool[i].source));
351  alGetError();
352  }
353  delete [] pool;
354  }
355 
356  bool getSource(TorcsSound* sound, ALuint* source, bool* needs_init, int* index) {
357  if (*index >= 0 && *index < nbsources) {
358  if (sound == pool[*index].currentOwner) {
359  // Resurrect source from pool.
360  *source = pool[*index].source;
361  *needs_init = false;
362  pool[*index].in_use = true;
363  return true;
364  }
365  }
366 
367  // TODO: Implement free list with ring buffer or whatever data structure
368  // to go from O(n) to O(1). If the ordering is done well it will automatically
369  // result in LRU (least recently used).
370  int i, firstfree = -1;
371  for (i = 0; i < nbsources; i++) {
372  if (pool[i].in_use == false && firstfree < 0) {
373  firstfree = i;
374  break;
375  }
376  }
377 
378  if (firstfree < 0) {
379  // No free source.
380  return false;
381  }
382 
383  pool[firstfree].currentOwner = sound;
384  pool[firstfree].in_use = true;
385  *source = pool[firstfree].source;
386  *needs_init = true;
387  *index = firstfree;
388  return true;
389  }
390 
391  bool releaseSource(TorcsSound* sound, int* index) {
392  if (*index >= 0 && *index < nbsources) {
393  if (sound == pool[*index].currentOwner) {
394  pool[*index].in_use = false;
395  return true;
396  }
397  }
398  return false;
399  }
400 
401  bool isSourceActive(TorcsSound* sound, int* index) {
402  if (*index >= 0 && *index < nbsources &&
403  sound == pool[*index].currentOwner &&
404  true == pool[*index].in_use)
405  {
406  return true;
407  } else {
408  return false;
409  }
410  }
411 
412  int getNbSources(void) { return nbsources; }
413 
414  protected:
415  int nbsources;
416  sharedSource *pool;
417 };
