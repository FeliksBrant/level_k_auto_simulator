157 tdble gfMean(tdble v, tMeanVal *pvt, int n, int w)
158 {
159  int i;
160  tdble sum;
161 
162  if (n > pvt->curNum) {
163  if (pvt->curNum < GF_MEAN_MAX_VAL) {
164  pvt->curNum++;
165  }
166  n = pvt->curNum;
167  } else {
168  pvt->curNum = n;
169  }
170 
171  sum = 0;
172  for (i = 0; i < n; i++) {
173  pvt->val[i] = pvt->val[i + 1];
174  sum += pvt->val[i];
175  }
176 
177  pvt->val[n] = v;
178  sum += (tdble)w * v;
179  sum /= (tdble)(n + w);
180 
181  return sum;
182 }
