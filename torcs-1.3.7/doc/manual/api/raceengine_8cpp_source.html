50 /* Compute Pit stop time */
51 static void
52 ReUpdtPitTime(tCarElt *car)
53 {
54  tSituation *s = ReInfo->s;
55  tReCarInfo *info = &(ReInfo->_reCarInfo[car->index]);
56  int i;
57 
58  switch (car->_pitStopType) {
59  case RM_PIT_REPAIR:
60  info->totalPitTime = ReInfo->raceRules.pitstopBaseTime + fabs((double)(car->_pitFuel)) / ReInfo->raceRules.refuelFuelFlow + (tdble)(fabs((double)(car->_pitRepair))) * ReInfo->raceRules.damageRepairFactor + car->_penaltyTime;
61  if (ReInfo->s->raceInfo.type == RM_TYPE_PRACTICE || ReInfo->s->raceInfo.type == RM_TYPE_QUALIF) {
62  // Ensure that the right min/max values are in the setup structure (could have been modified by the robot))
63  RtInitCarPitSetup(car->_carHandle, &(car->pitcmd.setup), true);
64  } else {
65  // In case of the race no modifications are allowed, so completely reload the structure
66  RtInitCarPitSetup(car->_carHandle, &(car->pitcmd.setup), false);
67  }
68  car->_scheduledEventTime = s->currentTime + info->totalPitTime;
69  car->_penaltyTime = 0.0f;
70  ReInfo->_reSimItf.reconfig(car);
71  for (i=0; i<4; i++) {
72  car->_tyreCondition(i) = 1.01;
73  car->_tyreT_in(i) = 50.0;
74  car->_tyreT_mid(i) = 50.0;
75  car->_tyreT_out(i) = 50.0;
76  }
77  break;
78  case RM_PIT_STOPANDGO:
79  info->totalPitTime = car->_penaltyTime;
80  car->_scheduledEventTime = s->currentTime + info->totalPitTime;
81  car->_penaltyTime = 0.0f;
82  break;
83  }
84 }

127 static void
128 ReManage(tCarElt *car)
129 {
130  int i, pitok;
131  tTrackSeg *sseg;
132  tdble wseg;
133  static float color[] = {0.0, 0.0, 1.0, 1.0};
134  tSituation *s = ReInfo->s;
135  const int BUFSIZE = 1024;
136  char buf[BUFSIZE];
137 
138  tReCarInfo *info = &(ReInfo->_reCarInfo[car->index]);
139 
140  if (car->_speed_x > car->_topSpeed) {
141  car->_topSpeed = car->_speed_x;
142  }
143 
144  // For practice and qualif.
145  if (car->_speed_x > info->topSpd) {
146  info->topSpd = car->_speed_x;
147  }
148 
149  if (car->_speed_x < info->botSpd) {
150  info->botSpd = car->_speed_x;
151  car->_currentMinSpeedForLap = car->_speed_x;
152  }
153 
154  // Pitstop.
155  if (car->_pit) {
156  if (car->ctrl.raceCmd & RM_CMD_PIT_ASKED) {
157  // Pit already occupied?
158  if (car->_pit->pitCarIndex == TR_PIT_STATE_FREE) {
159  snprintf(car->ctrl.msg[2], 32, "Can Pit");
160  } else {
161  snprintf(car->ctrl.msg[2], 32, "Pit Occupied");
162  }
163  memcpy(car->ctrl.msgColor, color, sizeof(car->ctrl.msgColor));
164  }
165 
166  if (car->_state & RM_CAR_STATE_PIT) {
167  car->ctrl.raceCmd &= ~RM_CMD_PIT_ASKED; // clear the flag.
168  if (car->_scheduledEventTime < s->currentTime) {
169  car->_state &= ~RM_CAR_STATE_PIT;
170  car->_pit->pitCarIndex = TR_PIT_STATE_FREE;
171  snprintf(buf, BUFSIZE, "%s pit stop %.1fs", car->_name, info->totalPitTime);
172  ReRaceMsgSet(buf, 5);
173  } else {
174  snprintf(car->ctrl.msg[2], 32, "in pits %.1fs", s->currentTime - info->startPitTime);
175  }
176  } else if ((car->ctrl.raceCmd & RM_CMD_PIT_ASKED) &&
177  car->_pit->pitCarIndex == TR_PIT_STATE_FREE &&
178  (s->_maxDammage == 0 || car->_dammage <= s->_maxDammage))
179  {
180  tdble lgFromStart = car->_trkPos.seg->lgfromstart;
181 
182  switch (car->_trkPos.seg->type) {
183  case TR_STR:
184  lgFromStart += car->_trkPos.toStart;
185  break;
186  default:
187  lgFromStart += car->_trkPos.toStart * car->_trkPos.seg->radius;
188  break;
189  }
190 
191  if ((lgFromStart > car->_pit->lmin) && (lgFromStart < car->_pit->lmax)) {
192  pitok = 0;
193  int side;
194  tdble toBorder;
195  if (ReInfo->track->pits.side == TR_RGT) {
196  side = TR_SIDE_RGT;
197  toBorder = car->_trkPos.toRight;
198  } else {
199  side = TR_SIDE_LFT;
200  toBorder = car->_trkPos.toLeft;
201  }
202 
203  sseg = car->_trkPos.seg->side[side];
204  wseg = RtTrackGetWidth(sseg, car->_trkPos.toStart);
205  if (sseg->side[side]) {
206  sseg = sseg->side[side];
207  wseg += RtTrackGetWidth(sseg, car->_trkPos.toStart);
208  }
209  if (((toBorder + wseg) < (ReInfo->track->pits.width - car->_dimension_y / 2.0)) &&
210  (fabs(car->_speed_x) < 1.0) &&
211  (fabs(car->_speed_y) < 1.0))
212  {
213  pitok = 1;
214  }
215 
216  if (pitok) {
217  car->_state |= RM_CAR_STATE_PIT;
218  car->_nbPitStops++;
219  for (i = 0; i < car->_pit->freeCarIndex; i++) {
220  if (car->_pit->car[i] == car) {
221  car->_pit->pitCarIndex = i;
222  break;
223  }
224  }
225  info->startPitTime = s->currentTime;
226  snprintf(buf, BUFSIZE, "%s in pits", car->_name);
227  ReRaceMsgSet(buf, 5);
228  if (car->robot->rbPitCmd(car->robot->index, car, s) == ROB_PIT_MENU) {
229  // the pit cmd is modified by menu.
230  ReStop();
231  RmPitMenuStart(car, ReInfo, (void*)car, ReUpdtPitCmd);
232  } else {
233  ReUpdtPitTime(car);
234  }
235  }
236  }
237  }
238  }
239 
240  /* Start Line Crossing */
241  if (info->prevTrkPos.seg != car->_trkPos.seg) {
242  if ((info->prevTrkPos.seg->raceInfo & TR_LAST) && (car->_trkPos.seg->raceInfo & TR_START)) {
243  if (info->lapFlag == 0) {
244  if ((car->_state & RM_CAR_STATE_FINISH) == 0) {
245  car->_laps++;
246  car->_remainingLaps--;
247  if (car->_laps > 1) {
248  car->_lastLapTime = s->currentTime - info->sTime;
249  car->_curTime += car->_lastLapTime;
250  if (car->_bestLapTime != 0) {
251  car->_deltaBestLapTime = car->_lastLapTime - car->_bestLapTime;
252  }
253  if ((car->_lastLapTime < car->_bestLapTime) || (car->_bestLapTime == 0)) {
254  if (car->_commitBestLapTime) {
255  car->_bestLapTime = car->_lastLapTime;
256  }
257  }
258 
259  car->_commitBestLapTime = true;
260 
261  if (car->_pos != 1) {
262  car->_timeBehindLeader = car->_curTime - s->cars[0]->_curTime;
263  car->_lapsBehindLeader = s->cars[0]->_laps - car->_laps;
264  car->_timeBehindPrev = car->_curTime - s->cars[car->_pos - 2]->_curTime;
265  s->cars[car->_pos - 2]->_timeBeforeNext = car->_timeBehindPrev;
266  } else {
267  car->_timeBehindLeader = 0;
268  car->_lapsBehindLeader = 0;
269  car->_timeBehindPrev = 0;
270 
271  if (ReInfo->_displayMode == RM_DISP_MODE_CONSOLE) {
272  printf("Sim Time: %8.2f [s], Leader Laps: %4d, Leader Distance: %8.3f [km]\n", s->currentTime, car->_laps - 1, car->_distRaced/1000.0f);
273  }
274  }
275  info->sTime = s->currentTime;
276  switch (ReInfo->s->_raceType) {
277  case RM_TYPE_PRACTICE:
278  if (ReInfo->_displayMode == RM_DISP_MODE_NONE) {
279  ReInfo->_refreshDisplay = 1;
280  const int TIMEFMTSIZE=256;
281  char t1[TIMEFMTSIZE], t2[TIMEFMTSIZE];
282  GfTime2Str(t1, TIMEFMTSIZE, car->_lastLapTime, 0);
283  GfTime2Str(t2, TIMEFMTSIZE, car->_bestLapTime, 0);
284  snprintf(buf, BUFSIZE, "lap: %02d time: %s best: %s top spd: %.2f min spd: %.2f damage: %d",
285  car->_laps - 1, t1, t2,
286  info->topSpd * 3.6, info->botSpd * 3.6, car->_dammage);
287  ReResScreenAddText(buf);
288  }
289  /* save the lap result */
290  ReSavePracticeLap(car);
291  break;
292 
293  case RM_TYPE_QUALIF:
294  if (ReInfo->_displayMode == RM_DISP_MODE_NONE) {
295  ReUpdateQualifCurRes(car);
296  }
297  break;
298  }
299  } else {
300  if ((ReInfo->_displayMode == RM_DISP_MODE_NONE) && (ReInfo->s->_raceType == RM_TYPE_QUALIF)) {
301  ReUpdateQualifCurRes(car);
302  }
303  }
304 
305  info->topSpd = car->_speed_x;
306  info->botSpd = car->_speed_x;
307  car->_currentMinSpeedForLap = car->_speed_x;
308  if ((car->_remainingLaps < 0) || (s->_raceState == RM_RACE_FINISHING)) {
309  car->_state |= RM_CAR_STATE_FINISH;
310  s->_raceState = RM_RACE_FINISHING;
311  if (ReInfo->s->_raceType == RM_TYPE_RACE) {
312  if (car->_pos == 1) {
313  snprintf(buf, BUFSIZE, "Winner %s", car->_name);
314  ReRaceBigMsgSet(buf, 10);
315  } else {
316  const char *numSuffix = "th";
317  if (abs(12 - car->_pos) > 1) { /* leave suffix as 'th' for 11 to 13 */
318  switch (car->_pos % 10) {
319  case 1:
320  numSuffix = "st";
321  break;
322  case 2:
323  numSuffix = "nd";
324  break;
325  case 3:
326  numSuffix = "rd";
327  break;
328  default:
329  break;
330  }
331  }
332  snprintf(buf, BUFSIZE, "%s Finished %d%s", car->_name, car->_pos, numSuffix);
333  ReRaceMsgSet(buf, 5);
334  }
335  }
336  }
337  } else {
338  /* prevent infinite looping of cars around track, allow one lap after finish for the first car */
339  for (i = 0; i < s->_ncars; i++) {
340  s->cars[i]->_state |= RM_CAR_STATE_FINISH;
341  }
342  return;
343  }
344  } else {
345  info->lapFlag--;
346  }
347  }
348 
349  if ((info->prevTrkPos.seg->raceInfo & TR_START) && (car->_trkPos.seg->raceInfo & TR_LAST)) {
350  /* going backward through the start line */
351  info->lapFlag++;
352  }
353  }
354  ReRaceRules(car);
355 
356  info->prevTrkPos = car->_trkPos;
357  car->_curLapTime = s->currentTime - info->sTime;
358  car->_distFromStartLine = car->_trkPos.seg->lgfromstart +
359  (car->_trkPos.seg->type == TR_STR ? car->_trkPos.toStart : car->_trkPos.toStart * car->_trkPos.seg->radius);
360  car->_distRaced = (car->_laps - (info->lapFlag + 1)) * ReInfo->track->length + car->_distFromStartLine;
361 }

364 static void ReSortCars(void)
365 {
366  int i, j;
367  tCarElt *car;
368  int allfinish;
369  tSituation *s = ReInfo->s;
370 
371  if ((s->cars[0]->_state & RM_CAR_STATE_FINISH) == 0) {
372  allfinish = 0;
373  } else {
374  allfinish = 1;
375  }
376 
377  for (i = 1; i < s->_ncars; i++) {
378  j = i;
379  while (j > 0) {
380  if ((s->cars[j]->_state & RM_CAR_STATE_FINISH) == 0) {
381  allfinish = 0;
382  if (s->cars[j]->_distRaced > s->cars[j-1]->_distRaced) {
383  car = s->cars[j];
384  s->cars[j] = s->cars[j-1];
385  s->cars[j-1] = car;
386  s->cars[j]->_pos = j+1;
387  s->cars[j-1]->_pos = j;
388  j--;
389  continue;
390  }
391  }
392  j = 0;
393  }
394  }
395 
396  if (allfinish) {
397  ReInfo->s->_raceState = RM_RACE_ENDED;
398  }
399 }

402 /* Compute the race rules and penalties */
403 static void
404 ReRaceRules(tCarElt *car)
405 {
406  tCarPenalty *penalty;
407  tTrack *track = ReInfo->track;
408  tRmCarRules *rules = &(ReInfo->rules[car->index]);
409  tTrackSeg *seg = RtTrackGetSeg(&(car->_trkPos));
410  tReCarInfo *info = &(ReInfo->_reCarInfo[car->index]);
411  tTrackSeg *prevSeg = RtTrackGetSeg(&(info->prevTrkPos));
412  static float color[] = {0.0, 0.0, 1.0, 1.0};
413 
414  // DNF cars which need too much time for the current lap, this is mainly to avoid
415  // that a "hanging" driver can stop the quali from finishing.
416  // Allowed time is longest pitstop possible + time for tracklength with speed??? (currently fixed 10 [m/s]).
417  // for simplicity. Human driver is an exception to this rule, to allow explorers
418  // to enjoy the landscape.
419  // TODO: Make it configurable.
420  if ((car->_curLapTime > 84.5 + ReInfo->track->length/10.0) &&
421  (car->_driverType != RM_DRV_HUMAN))
422  {
423  car->_state |= RM_CAR_STATE_ELIMINATED;
424  return;
425  }
426 
427  const int BUFSIZE = 1024;
428  char buf[BUFSIZE];
429 
430  // Ignore some rules after the car has finished the race
431  if ((car->pub.state & RM_CAR_STATE_FINISH) == 0) {
432 
433  // If a car hits the track wall the lap time is invalidated, because of tracks where this behaviour allows much faster laps (e.g. alpine-2)
434  // Invalidation and message is just shown on the first hit
435  if (ReInfo->raceRules.enabled & RmRaceRules::WALL_HIT_TIME_INVALIDATE) {
436  if (car->_commitBestLapTime && (car->priv.simcollision & SEM_COLLISION_XYSCENE)) {
437  car->_commitBestLapTime = false;
438  if (ReInfo->s->_raceType != RM_TYPE_RACE) {
439  ReRaceMsgSet("Hit wall, laptime invalidated", 5);
440  }
441  }
442  }
443 
444  // If the car cuts a corner the lap time is invalidated. Cutting a corner means: the center of gravity is more than 0.7 times the car width
445  // away from the main track segment on the inside of a turn. The rule does not apply on the outside and on straights, pit entry and exit
446  // count as well as track.
447  tTrackSeg *mainseg = car->_trkPos.seg;
448  bool pit = false;
449  tTrackPitInfo pitInfo = track->pits;
450  tdble toborder = 0.0f;
451  tdble minradius = 1.0f;
452 
453  if (mainseg->type != TR_STR) {
454  if (track->pits.type == TR_PIT_ON_TRACK_SIDE) {
455  if (pitInfo.pitEntry->id < pitInfo.pitExit->id) {
456  if ((mainseg->id >= pitInfo.pitEntry->id) && (mainseg->id <= pitInfo.pitExit->id)) {
457  pit = true;
458  }
459  } else {
460  if ((mainseg->id >= pitInfo.pitEntry->id) || (mainseg->id <= pitInfo.pitExit->id)) {
461  pit = true;
462  }
463  }
464  }
465 
466  if (mainseg->type == TR_LFT) {
467  if (!(pit && (pitInfo.side == TR_LFT))) {
468  toborder = car->_trkPos.toLeft;
469  minradius = mainseg->radiusl;
470  }
471  } else if (mainseg->type == TR_RGT) {
472  if (!(pit && (pitInfo.side == TR_RGT))) {
473  toborder = car->_trkPos.toRight;
474  minradius = mainseg->radiusr;
475  }
476  }
477  }
478 
479  tdble cuttinglimit = car->_dimension_y*0.7f;
480  if (toborder < -cuttinglimit) {
481  if (ReInfo->raceRules.enabled & RmRaceRules::CORNER_CUTTING_TIME_INVALIDATE) {
482  if (ReInfo->s->_raceType != RM_TYPE_RACE && car->_commitBestLapTime) {
483  ReRaceMsgSet("Cut corner, laptime invalidated", 5);
484  }
485  car->_commitBestLapTime = false;
486  }
487  if (ReInfo->s->_raceType == RM_TYPE_RACE && ReInfo->raceRules.enabled & RmRaceRules::CORNER_CUTTING_TIME_PENALTY) {
488  // In race, apply additionally corner cutting time penalty
489  minradius -= cuttinglimit;
490  if (minradius > 1.0f) {
491  car->_penaltyTime += car->pub.speed*RCM_MAX_DT_SIMU*(-toborder-cuttinglimit)/minradius;
492  }
493  }
494  }
495  }
496 
497  if (car->_skillLevel < 3) {
498  /* only for the pros */
499  return;
500  }
501 
502  penalty = GF_TAILQ_FIRST(&(car->_penaltyList));
503  if (penalty) {
504  if (car->_laps > penalty->lapToClear) {
505  /* too late to clear the penalty, out of race */
506  car->_state |= RM_CAR_STATE_ELIMINATED;
507  return;
508  }
509 
510  switch (penalty->penalty) {
511  case RM_PENALTY_DRIVETHROUGH:
512  snprintf(car->ctrl.msg[3], 32, "Drive Through Penalty");
513  break;
514  case RM_PENALTY_STOPANDGO:
515  snprintf(car->ctrl.msg[3], 32, "Stop And Go Penalty");
516  break;
517  default:
518  *(car->ctrl.msg[3]) = 0;
519  break;
520  }
521 
522  memcpy(car->ctrl.msgColor, color, sizeof(car->ctrl.msgColor));
523  }
524 
525  if (prevSeg->raceInfo & TR_PITSTART) {
526  /* just entered the pit lane */
527  if (seg->raceInfo & TR_PIT) {
528  /* may be a penalty can be cleaned up */
529  if (penalty) {
530  switch (penalty->penalty) {
531  case RM_PENALTY_DRIVETHROUGH:
532  snprintf(buf, BUFSIZE, "%s DRIVE THROUGH PENALTY CLEANING", car->_name);
533  ReRaceMsgSet(buf, 5);
534  rules->ruleState |= RM_PNST_DRIVETHROUGH;
535  break;
536  case RM_PENALTY_STOPANDGO:
537  snprintf(buf, BUFSIZE, "%s STOP&GO PENALTY CLEANING", car->_name);
538  ReRaceMsgSet(buf, 5);
539  rules->ruleState |= RM_PNST_STOPANDGO;
540  break;
541  }
542  }
543  }
544  } else if (prevSeg->raceInfo & TR_PIT) {
545  if (seg->raceInfo & TR_PIT) {
546  /* the car stopped in pits */
547  if (car->_state & RM_CAR_STATE_PIT) {
548  if (rules->ruleState & RM_PNST_DRIVETHROUGH) {
549  /* it's not more a drive through */
550  rules->ruleState &= ~RM_PNST_DRIVETHROUGH;
551  } else if (rules->ruleState & RM_PNST_STOPANDGO) {
552  rules->ruleState |= RM_PNST_STOPANDGO_OK;
553  }
554  } else {
555  if(rules->ruleState & RM_PNST_STOPANDGO_OK && car->_pitStopType != RM_PIT_STOPANDGO) {
556  rules->ruleState &= ~ ( RM_PNST_STOPANDGO | RM_PNST_STOPANDGO_OK );
557  }
558  }
559  } else if (seg->raceInfo & TR_PITEND) {
560  /* went out of the pit lane, check if the current penalty is cleared */
561  if (rules->ruleState & (RM_PNST_DRIVETHROUGH | RM_PNST_STOPANDGO_OK)) {
562  /* clear the penalty */
563  snprintf(buf, BUFSIZE, "%s penalty cleared", car->_name);
564  ReRaceMsgSet(buf, 5);
565  penalty = GF_TAILQ_FIRST(&(car->_penaltyList));
566  GF_TAILQ_REMOVE(&(car->_penaltyList), penalty, link);
567  FREEZ(penalty);
568  }
569 
570  rules->ruleState = 0;
571  } else {
572  /* went out of the pit lane illegally... */
573  /* it's a new stop and go... */
574  if (!(rules->ruleState & RM_PNST_STNGO)) {
575  snprintf(buf, BUFSIZE, "%s STOP&GO PENALTY", car->_name);
576  ReRaceMsgSet(buf, 5);
577  penalty = (tCarPenalty*)calloc(1, sizeof(tCarPenalty));
578  penalty->penalty = RM_PENALTY_STOPANDGO;
579  penalty->lapToClear = car->_laps + 5;
580  GF_TAILQ_INSERT_TAIL(&(car->_penaltyList), penalty, link);
581  rules->ruleState = RM_PNST_STNGO;
582  }
583  }
584  } else if (seg->raceInfo & TR_PITEND) {
585  rules->ruleState = 0;
586  } else if (seg->raceInfo & TR_PIT) {
587  /* entrered the pits not from the pit entry... */
588  /* it's a new stop and go... */
589  if (!(rules->ruleState & RM_PNST_STNGO)) {
590  snprintf(buf, BUFSIZE, "%s STOP&GO PENALTY", car->_name);
591  ReRaceMsgSet(buf, 5);
592  penalty = (tCarPenalty*)calloc(1, sizeof(tCarPenalty));
593  penalty->penalty = RM_PENALTY_STOPANDGO;
594  penalty->lapToClear = car->_laps + 5;

614 static void
615 ReOneStep(double deltaTimeIncrement)
616 {
617  int i;
618  tRobotItf *robot;
619  tSituation *s = ReInfo->s;
620 
621  if ((ReInfo->_displayMode != RM_DISP_MODE_NONE) && (ReInfo->_displayMode != RM_DISP_MODE_CONSOLE)) {
622  if (floor(s->currentTime) == -2.0) {
623  ReRaceBigMsgSet("Ready", 1.0);
624  } else if (floor(s->currentTime) == -1.0) {
625  ReRaceBigMsgSet("Set", 1.0);
626  } else if (floor(s->currentTime) == 0.0) {
627  ReRaceBigMsgSet("Go", 1.0);
628  }
629  }
630 
631  ReInfo->_reCurTime += deltaTimeIncrement * ReInfo->_reTimeMult; /* "Real" time */
632  s->currentTime += deltaTimeIncrement; /* Simulated time */
633 
634  if (s->currentTime < 0) {
635  /* no simu yet */
636  ReInfo->s->_raceState = RM_RACE_PRESTART;
637  } else if (ReInfo->s->_raceState == RM_RACE_PRESTART) {
638  ReInfo->s->_raceState = RM_RACE_RUNNING;
639  s->currentTime = 0.0; /* resynchronize */
640  ReInfo->_reLastTime = 0.0;
641  }
642 
643  START_PROFILE("rbDrive*");
644  if ((s->currentTime - ReInfo->_reLastTime) >= RCM_MAX_DT_ROBOTS) {
645  s->deltaTime = s->currentTime - ReInfo->_reLastTime;
646  for (i = 0; i < s->_ncars; i++) {
647  if ((s->cars[i]->_state & RM_CAR_STATE_NO_SIMU) == 0) {
648  robot = s->cars[i]->robot;
649  robot->rbDrive(robot->index, s->cars[i], s);
650  }
651  }
652  ReInfo->_reLastTime = s->currentTime;
653  }
654  STOP_PROFILE("rbDrive*");
655 
656  START_PROFILE("_reSimItf.update*");
657  ReInfo->_reSimItf.update(s, deltaTimeIncrement, -1);
658  for (i = 0; i < s->_ncars; i++) {
659  ReManage(s->cars[i]);
660  }
661  STOP_PROFILE("_reSimItf.update*");
662 
663  if ((ReInfo->_displayMode != RM_DISP_MODE_NONE) && (ReInfo->_displayMode != RM_DISP_MODE_CONSOLE)) {
664  ReRaceMsgUpdate();
665  }
666  ReSortCars();
667 }

709 int
710 ReUpdate(void)
711 {
712  double t;
713  tRmMovieCapture *capture;
714  int mode = RM_ASYNC;
715  int i;
716  const int MAXSTEPS = 2000;
717 
718  START_PROFILE("ReUpdate");
719  ReInfo->_refreshDisplay = 0;
720  switch (ReInfo->_displayMode) {
721  case RM_DISP_MODE_NORMAL:
722  t = GfTimeClock();
723 
724  i = 0;
725  START_PROFILE("ReOneStep*");
726  while ((ReInfo->_reRunning && ((t - ReInfo->_reCurTime) > RCM_MAX_DT_SIMU)) && MAXSTEPS > i++) {
727  ReOneStep(RCM_MAX_DT_SIMU);
728  }
729  STOP_PROFILE("ReOneStep*");
730 
731  if (i > MAXSTEPS) {
732  // Cannot keep up with time warp, reset time to avoid lag when running slower again
733  ReInfo->_reCurTime = GfTimeClock();
734  }
735 
736  GfuiDisplay();
737  ReInfo->_reGraphicItf.refresh(ReInfo->s);
738  glutPostRedisplay(); /* Callback -> reDisplay */
739  break;
740 
741  case RM_DISP_MODE_NONE:
742  // Just update view once per 2 seconds simulation time to avoid trouble with graphics cards
743  // which are bad in buffer switching (e.g. ATI fglrx driver on Linux).
744  t = ReInfo->_reCurTime;
745  while ((t - ReInfo->_reCurTime + 2.0) > 0.0) {
746  ReOneStep(RCM_MAX_DT_SIMU);
747  }
748 
749  GfuiDisplay();
750  glutPostRedisplay(); /* Callback -> reDisplay */
751  break;
752 
753  case RM_DISP_MODE_CAPTURE:
754  capture = &(ReInfo->movieCapture);
755  while ((ReInfo->_reCurTime - capture->lastFrame) < capture->deltaFrame) {
756  ReOneStep(capture->deltaSimu);
757  }
758  capture->lastFrame = ReInfo->_reCurTime;
759 
760  GfuiDisplay();
761  ReInfo->_reGraphicItf.refresh(ReInfo->s);
762  reCapture();
763  glutPostRedisplay(); /* Callback -> reDisplay */
764  break;
765 
766  case RM_DISP_MODE_CONSOLE:
767  t = ReInfo->_reCurTime;
768  while ((t - ReInfo->_reCurTime + 2.0) > 0.0) {
769  ReOneStep(RCM_MAX_DT_SIMU);
770  }
771  mode = RM_SYNC;
772  break;
773 
774  }
775  STOP_PROFILE("ReUpdate");
776 
777  return mode;
778 }
