45 static void
46 ctrlCheck(tCar *car)
47 {
48  tTransmission *trans = &(car->transmission);
49  tClutch *clutch = &(trans->clutch);
50 
51  /* sanity check */
52 #if defined WIN32 || defined sun
53  if (isnan(car->ctrl->accelCmd)) car->ctrl->accelCmd = 0;
54  if (isnan(car->ctrl->brakeCmd)) car->ctrl->brakeCmd = 0;
55  if (isnan(car->ctrl->clutchCmd)) car->ctrl->clutchCmd = 0;
56  if (isnan(car->ctrl->steer)) car->ctrl->steer = 0;
57 #else
58  if (isnan(car->ctrl->accelCmd) || isinf(car->ctrl->accelCmd)) car->ctrl->accelCmd = 0;
59  if (isnan(car->ctrl->brakeCmd) || isinf(car->ctrl->brakeCmd)) car->ctrl->brakeCmd = 0;
60  if (isnan(car->ctrl->clutchCmd) || isinf(car->ctrl->clutchCmd)) car->ctrl->clutchCmd = 0;
61  if (isnan(car->ctrl->steer) || isinf(car->ctrl->steer)) car->ctrl->steer = 0;
62 #endif
63 
64  /* When the car is broken try to send it on the track side */
65  if (car->carElt->_state & RM_CAR_STATE_BROKEN) {
66  car->ctrl->accelCmd = 0.0f;
67  car->ctrl->brakeCmd = 0.1f;
68  car->ctrl->gear = 0;
69  if (car->trkPos.toRight > car->trkPos.seg->width / 2.0) {
70  car->ctrl->steer = 0.1f;
71  } else {
72  car->ctrl->steer = -0.1f;
73  }
74  } else if (car->carElt->_state & RM_CAR_STATE_ELIMINATED) {
75  car->ctrl->accelCmd = 0.0f;
76  car->ctrl->brakeCmd = 0.1f;
77  car->ctrl->gear = 0;
78  if (car->trkPos.toRight > car->trkPos.seg->width / 2.0) {
79  car->ctrl->steer = 0.1f;
80  } else {
81  car->ctrl->steer = -0.1f;
82  }
83  } else if (car->carElt->_state & RM_CAR_STATE_FINISH) {
84  /* when the finish line is passed, continue at "slow" pace */
85  car->ctrl->accelCmd = MIN(car->ctrl->accelCmd, 0.20);
86  if (car->DynGC.vel.x > 30.0) {
87  car->ctrl->brakeCmd = MAX(car->ctrl->brakeCmd, 0.05);
88  }
89  }
90 
91  /* check boundaries */
92  if (car->ctrl->accelCmd > 1.0) {
93  car->ctrl->accelCmd = 1.0;
94  } else if (car->ctrl->accelCmd < 0.0) {
95  car->ctrl->accelCmd = 0.0;
96  }
97  if (car->ctrl->brakeCmd > 1.0) {
98  car->ctrl->brakeCmd = 1.0;
99  } else if (car->ctrl->brakeCmd < 0.0) {
100  car->ctrl->brakeCmd = 0.0;
101  }
102  if (car->ctrl->clutchCmd > 1.0) {
103  car->ctrl->clutchCmd = 1.0;
104  } else if (car->ctrl->clutchCmd < 0.0) {
105  car->ctrl->clutchCmd = 0.0;
106  }
107  if (car->ctrl->steer > 1.0) {
108  car->ctrl->steer = 1.0;
109  } else if (car->ctrl->steer < -1.0) {
110  car->ctrl->steer = -1.0;
111  }
112 
113  clutch->transferValue = 1.0 - car->ctrl->clutchCmd;
114 }

168 static void
169 RemoveCar(tCar *car, tSituation *s)
170 {
171  int i;
172  tCarElt *carElt;
173  tTrkLocPos trkPos;
174  int trkFlag;
175  tdble travelTime;
176  tdble dang;
177 
178  static tdble PULL_Z_OFFSET = 3.0;
179  static tdble PULL_SPD = 0.5;
180 
181  carElt = car->carElt;
182 
183  if (carElt->_state & RM_CAR_STATE_PULLUP) {
184  carElt->_pos_Z += car->restPos.vel.z * SimDeltaTime;
185  carElt->_yaw += car->restPos.vel.az * SimDeltaTime;
186  carElt->_roll += car->restPos.vel.ax * SimDeltaTime;
187  carElt->_pitch += car->restPos.vel.ay * SimDeltaTime;
188  sgMakeCoordMat4(carElt->pub.posMat, carElt->_pos_X - carElt->_statGC_x, carElt->_pos_Y - carElt->_statGC_y, carElt->_pos_Z - carElt->_statGC_z,
189  RAD2DEG(carElt->_yaw), RAD2DEG(carElt->_roll), RAD2DEG(carElt->_pitch));
190 
191  if (carElt->_pos_Z > (car->restPos.pos.z + PULL_Z_OFFSET)) {
192  carElt->_state &= ~RM_CAR_STATE_PULLUP;
193  carElt->_state |= RM_CAR_STATE_PULLSIDE;
194  // Moved pullside velocity computation down due to floating point error accumulation.
195  }
196  return;
197  }
198 
199 
200  if (carElt->_state & RM_CAR_STATE_PULLSIDE) {
201  // Recompute speed to avoid missing the parking point due to error accumulation (the pos might be
202  // in the 0-10000 range, depending on the track and vel*dt is around 0-0.001, so basically all
203  // but the most significant digits are lost under bad conditions, happens e.g on e-track-4).
204  // Should not lead to a division by zero because the pullside process stops if the car is within
205  // [0.5, 0.5]. Do not move it back.
206  travelTime = DIST(car->restPos.pos.x, car->restPos.pos.y, carElt->_pos_X, carElt->_pos_Y) / PULL_SPD;
207  car->restPos.vel.x = (car->restPos.pos.x - carElt->_pos_X) / travelTime;
208  car->restPos.vel.y = (car->restPos.pos.y - carElt->_pos_Y) / travelTime;
209 
210  carElt->_pos_X += car->restPos.vel.x * SimDeltaTime;
211  carElt->_pos_Y += car->restPos.vel.y * SimDeltaTime;
212  sgMakeCoordMat4(carElt->pub.posMat, carElt->_pos_X - carElt->_statGC_x, carElt->_pos_Y - carElt->_statGC_y, carElt->_pos_Z - carElt->_statGC_z,
213  RAD2DEG(carElt->_yaw), RAD2DEG(carElt->_roll), RAD2DEG(carElt->_pitch));
214 
215  if ((fabs(car->restPos.pos.x - carElt->_pos_X) < 0.5) && (fabs(car->restPos.pos.y - carElt->_pos_Y) < 0.5)) {
216  carElt->_state &= ~RM_CAR_STATE_PULLSIDE;
217  carElt->_state |= RM_CAR_STATE_PULLDN;
218  }
219  return;
220  }
221 
222 
223  if (carElt->_state & RM_CAR_STATE_PULLDN) {
224  carElt->_pos_Z -= car->restPos.vel.z * SimDeltaTime;
225  sgMakeCoordMat4(carElt->pub.posMat, carElt->_pos_X - carElt->_statGC_x, carElt->_pos_Y - carElt->_statGC_y, carElt->_pos_Z - carElt->_statGC_z,
226  RAD2DEG(carElt->_yaw), RAD2DEG(carElt->_roll), RAD2DEG(carElt->_pitch));
227 
228  if (carElt->_pos_Z < car->restPos.pos.z) {
229  carElt->_state &= ~RM_CAR_STATE_PULLDN;
230  carElt->_state |= RM_CAR_STATE_OUT;
231  }
232  return;
233  }
234 
235 
236  if (carElt->_state & (RM_CAR_STATE_NO_SIMU & ~RM_CAR_STATE_PIT)) {
237  return;
238  }
239 
240  if (carElt->_state & RM_CAR_STATE_PIT) {
241  if ((s->_maxDammage) && (car->dammage > s->_maxDammage)) {
242  // Broken during pit stop.
243  carElt->_state &= ~RM_CAR_STATE_PIT;
244  carElt->_pit->pitCarIndex = TR_PIT_STATE_FREE;
245  } else {
246  return;
247  }
248  }
249 
250  if ((s->_maxDammage) && (car->dammage > s->_maxDammage)) {
251  carElt->_state |= RM_CAR_STATE_BROKEN;
252  } else {
253  carElt->_state |= RM_CAR_STATE_OUTOFGAS;
254  }
255 
256  carElt->_gear = car->transmission.gearbox.gear = 0;
257  carElt->_enginerpm = car->engine.rads = 0;
258 
259  if (!(carElt->_state & RM_CAR_STATE_DNF)) {
260  if (fabs(carElt->_speed_x) > 1.0) {
261  return;
262  }
263  }
264 
265  carElt->_state |= RM_CAR_STATE_PULLUP;
266  // RM_CAR_STATE_NO_SIMU evaluates to > 0 from here, so we remove the car from the
267  // collision detection.
268  SimCollideRemoveCar(car, s->_ncars);
269 
270  carElt->priv.simcollision = carElt->priv.collision = car->collision = 0;
271  for(i = 0; i < 4; i++) {
272  carElt->_skid[i] = 0;
273  carElt->_wheelSpinVel(i) = 0;
274  carElt->_brakeTemp(i) = 0;
275  }
276 
277  carElt->pub.DynGC = car->DynGC;
278  carElt->_speed_x = 0;
279 
280  // Compute the target zone for the wrecked car.
281  trkPos = car->trkPos;
282  if (trkPos.toRight > trkPos.seg->width / 2.0) {
283  while (trkPos.seg->lside != 0) {
284  trkPos.seg = trkPos.seg->lside;
285  }
286  trkPos.toLeft = -3.0;
287  trkFlag = TR_TOLEFT;
288  } else {
289  while (trkPos.seg->rside != 0) {
290  trkPos.seg = trkPos.seg->rside;
291  }
292  trkPos.toRight = -3.0;
293  trkFlag = TR_TORIGHT;
294  }
295 
296  trkPos.type = TR_LPOS_SEGMENT;
297  RtTrackLocal2Global(&trkPos, &(car->restPos.pos.x), &(car->restPos.pos.y), trkFlag);
298  car->restPos.pos.z = RtTrackHeightL(&trkPos) + carElt->_statGC_z;
299  car->restPos.pos.az = RtTrackSideTgAngleL(&trkPos);
300  car->restPos.pos.ax = 0;
301  car->restPos.pos.ay = 0;
302 
303  car->restPos.vel.z = PULL_SPD;
304  travelTime = (car->restPos.pos.z + PULL_Z_OFFSET - carElt->_pos_Z) / car->restPos.vel.z;
305  dang = car->restPos.pos.az - carElt->_yaw;
306  NORM_PI_PI(dang);
307  car->restPos.vel.az = dang / travelTime;
308  dang = car->restPos.pos.ax - carElt->_roll;
309  NORM_PI_PI(dang);
310  car->restPos.vel.ax = dang / travelTime;
311  dang = car->restPos.pos.ay - carElt->_pitch;
312  NORM_PI_PI(dang);
313  car->restPos.vel.ay = dang / travelTime;
314 }

318 void
319 SimUpdate(tSituation *s, double deltaTime, int telemetry)
320 {
321  int i;
322  int ncar;
323  tCarElt *carElt;
324  tCar *car;
325 
326  SimDeltaTime = deltaTime;
327  SimTelemetry = telemetry;
328  for (ncar = 0; ncar < s->_ncars; ncar++) {
329  SimCarTable[ncar].collision = 0;
330  SimCarTable[ncar].blocked = 0;
331  }
332 
333  for (ncar = 0; ncar < s->_ncars; ncar++) {
334  car = &(SimCarTable[ncar]);
335  carElt = car->carElt;
336 
337  if (carElt->_state & RM_CAR_STATE_NO_SIMU) {
338  RemoveCar(car, s);
339  continue;
340  } else if (((s->_maxDammage) && (car->dammage > s->_maxDammage)) ||
341  (car->fuel == 0) ||
342  (car->carElt->_state & RM_CAR_STATE_ELIMINATED)) {
343  RemoveCar(car, s);
344  if (carElt->_state & RM_CAR_STATE_NO_SIMU) {
345  continue;
346  }
347  }
348 
349  if (s->_raceState & RM_RACE_PRESTART) {
350  car->ctrl->gear = 0;
351  }
352 
353  CHECK(car);
354  ctrlCheck(car);
355  CHECK(car);
356  SimSteerUpdate(car);
357  CHECK(car);
358  SimGearboxUpdate(car);
359  CHECK(car);
360  SimEngineUpdateTq(car);
361  CHECK(car);
362 
363  if (!(s->_raceState & RM_RACE_PRESTART)) {
364 
365  SimCarUpdateWheelPos(car);
366  CHECK(car);
367  SimBrakeSystemUpdate(car);
368  CHECK(car);
369  SimAeroUpdate(car, s);
370  CHECK(car);
371  for (i = 0; i < 2; i++){
372  SimWingUpdate(car, i, s);
373  }
374  CHECK(car);
375  for (i = 0; i < 4; i++){
376  SimWheelUpdateRide(car, i);
377  }
378  CHECK(car);
379  for (i = 0; i < 2; i++){
380  SimAxleUpdate(car, i);
381  }
382  CHECK(car);
383  for (i = 0; i < 4; i++){
384  SimWheelUpdateForce(car, i);
385  }
386  CHECK(car);
387  SimTransmissionUpdate(car);
388  CHECK(car);
389  SimWheelUpdateRotation(car);
390  CHECK(car);
391  SimCarUpdate(car, s);
392  CHECK(car);
393  } else {
394  SimEngineUpdateRpm(car, 0.0);
395  }
396  }
397 
398  SimCarCollideCars(s);
399 
400  /* printf ("%f - ", s->currentTime); */
401 
402  for (ncar = 0; ncar < s->_ncars; ncar++) {
403  car = &(SimCarTable[ncar]);
404  CHECK(car);
405  carElt = car->carElt;
406 
407  if (carElt->_state & RM_CAR_STATE_NO_SIMU) {
408  continue;
409  }
410 
411  CHECK(car);
412  SimCarUpdate2(car, s); /* telemetry */
413 
414  /* copy back the data to carElt */
415 
416  carElt->pub.DynGC = car->DynGC;
417  carElt->pub.DynGCg = car->DynGCg;
418  carElt->pub.speed = car->speed;
419  sgMakeCoordMat4(carElt->pub.posMat, carElt->_pos_X - carElt->_statGC_x, carElt->_pos_Y - carElt->_statGC_y, carElt->_pos_Z - carElt->_statGC_z,
420  RAD2DEG(carElt->_yaw), RAD2DEG(carElt->_roll), RAD2DEG(carElt->_pitch));
421  carElt->_trkPos = car->trkPos;
422  for (i = 0; i < 4; i++) {
423  carElt->priv.wheel[i].relPos = car->wheel[i].relPos;
424  carElt->_wheelSeg(i) = car->wheel[i].trkPos.seg;
425  carElt->_brakeTemp(i) = car->wheel[i].brake.temp;
426  carElt->pub.corner[i] = car->corner[i].pos;
427  }
428  carElt->_gear = car->transmission.gearbox.gear;
429  carElt->_enginerpm = car->engine.rads;
430  carElt->_fuel = car->fuel;
431  carElt->priv.collision |= car->collision;
432  carElt->priv.simcollision = car->collision;
433  carElt->_dammage = car->dammage;
434  }
435 }
